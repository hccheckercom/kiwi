## Bad
```ts
export async function generateDescription(product: Product) {
  const response = await anthropic.messages.create({
    model: 'claude-sonnet-4-6',
    messages: [{ role: 'user', content: `Describe: ${product.name}` }]
  })
  return response.content[0].text
}
// If Anthropic is down → entire product creation flow fails
```

## Good
```ts
export async function generateDescription(product: Product) {
  try {
    const response = await anthropic.messages.create({
      model: 'claude-sonnet-4-6',
      max_tokens: 512,
      messages: [{ role: 'user', content: `Describe: ${product.name}` }]
    })
    return response.content[0].text
  } catch (err) {
    console.error('AI generation failed, using template fallback', err)
    // Graceful degradation — feature works without AI
    return `${product.name} - ${product.category}. Liên hệ để biết thêm chi tiết.`
  }
}
```

## Why
AI APIs have ~99.5% uptime — that's 44 hours of downtime per year. If your feature has no fallback, it's completely broken during outages. Non-critical AI features (descriptions, suggestions, summaries) should degrade gracefully to templates or cached results.
