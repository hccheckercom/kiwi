```php
<!-- wallet.php -->
<?php
declare(strict_types=1);

if ( ! defined( 'ABSPATH' ) ) { exit; }

if ( ! is_user_logged_in() ) {
    wp_safe_redirect( home_url( '/my-account/login/' ) );
    exit;
}

get_header();

$user_id      = get_current_user_id();
$balance      = function_exists( 'wz_get_wallet_balance' ) ? wz_get_wallet_balance( $user_id ) : 0;
$transactions = function_exists( 'wz_get_wallet_transactions' ) ? wz_get_wallet_transactions( $user_id, 1 ) : [];

set_query_var( 'wz_tpl_account_page', 'dashboard' );
?>

<div class="bg-[var(--wz-surface)] min-h-screen">
<div class="max-w-[1280px] mx-auto px-5 md:px-[20px] py-4">

    <nav aria-label="Breadcrumb" class="pb-4">
        <ol class="flex items-center gap-1 text-xs md:text-sm text-[var(--wz-outline)]">
            <li><a href="<?php echo esc_url( home_url( '/' ) ); ?>" class="hover:text-[var(--wz-primary)] transition">Trang chủ</a></li>
            <li><span class="material-symbols-outlined text-[14px]" aria-hidden="true">chevron_right</span></li>
            <li><a href="<?php echo esc_url( home_url( '/my-account/' ) ); ?>" class="hover:text-[var(--wz-primary)] transition">Tài khoản</a></li>
            <li><span class="material-symbols-outlined text-[14px]" aria-hidden="true">chevron_right</span></li>
            <li class="font-bold text-[var(--wz-primary)]" aria-current="page">Ví của tôi</li>
        </ol>
    </nav>

    <div class="flex flex-col lg:flex-row lg:gap-6">
        <?php include get_stylesheet_directory() . '/template-parts/account/sidebar.php'; ?>

        <div class="flex-1 min-w-0 mt-4 lg:mt-0 space-y-4">
            <!-- Balance Card -->
            <div class="bg-gradient-to-r from-[var(--wz-primary)] to-[var(--wz-primary)]/80 rounded-[var(--wz-radius-card)] p-6 text-white">
                <p class="text-sm text-white/70">Số dư hiện tại</p>
                <p class="text-3xl font-bold mt-1"><?php echo esc_html( wz_format_price( (float) $balance ) ); ?></p>
            </div>

            <!-- Transaction History -->
            <div class="bg-[var(--wz-surface-pure)] rounded-[var(--wz-radius-card)] p-5 shadow-[0_4px_24px_rgba(27,58,122,0.08)]">
                <h2 class="text-base font-bold text-[var(--wz-primary)] mb-4">Lịch sử giao dịch</h2>

                <div class="flex gap-2 overflow-x-auto pb-2 mb-4">
                    <button type="button" class="wz-wallet-tab bg-[var(--wz-primary)] text-white px-3 py-1.5 rounded-full text-xs font-semibold" data-filter="all">Tất cả</button>
                    <button type="button" class="wz-wallet-tab bg-[var(--wz-surface-cool)] text-[var(--wz-primary)] px-3 py-1.5 rounded-full text-xs font-semibold" data-filter="topup">Nạp tiền</button>
                    <button type="button" class="wz-wallet-tab bg-[var(--wz-surface-cool)] text-[var(--wz-primary)] px-3 py-1.5 rounded-full text-xs font-semibold" data-filter="spend">Sử dụng</button>
                    <button type="button" class="wz-wallet-tab bg-[var(--wz-surface-cool)] text-[var(--wz-primary)] px-3 py-1.5 rounded-full text-xs font-semibold" data-filter="refund">Hoàn tiền</button>
                </div>

                <?php if ( empty( $transactions ) ) : ?>
                    <div class="text-center py-8">
                        <span class="material-symbols-outlined text-[48px] text-[var(--wz-outline)]/30" aria-hidden="true">account_balance_wallet</span>
                        <p class="text-sm text-[var(--wz-outline)] mt-2">Chưa có giao dịch nào</p>
                        <a href="<?php echo esc_url( wz_get_shop_url() ); ?>" class="inline-flex items-center gap-2 mt-4 px-5 py-2 bg-[var(--wz-primary)] text-white rounded-[var(--wz-radius)] text-sm font-semibold hover:brightness-110 transition">Bắt đầu mua sắm</a>
                    </div>
                <?php else : ?>
                    <div class="space-y-0">
                        <?php
                        $type_icons = [
                            'topup'  => [ 'icon' => 'add_circle',    'bg' => 'bg-[var(--wz-accent)]/10',  'color' => 'text-[var(--wz-accent)]' ],
                            'spend'  => [ 'icon' => 'remove_circle', 'bg' => 'bg-[var(--wz-error)]/10',   'color' => 'text-[var(--wz-error)]' ],
                            'refund' => [ 'icon' => 'refresh',       'bg' => 'bg-[var(--wz-primary)]/10', 'color' => 'text-[var(--wz-primary)]' ],
                        ];
                        foreach ( $transactions as $tx ) :
                            $type = $tx['type'] ?? 'spend';
                            $icons = $type_icons[ $type ] ?? $type_icons['spend'];
                            $is_positive = in_array( $type, [ 'topup', 'refund' ], true );
                        ?>
                            <div class="wz-wallet-tx flex items-center gap-3 py-3 border-b border-[var(--wz-outline)]/10 last:border-0" data-type="<?php echo esc_attr( $type ); ?>">
                                <div class="w-10 h-10 rounded-full flex items-center justify-center <?php echo esc_attr( $icons['bg'] ); ?>">
                                    <span class="material-symbols-outlined text-[20px] <?php echo esc_attr( $icons['color'] ); ?>" aria-hidden="true"><?php echo esc_html( $icons['icon'] ); ?></span>
                                </div>
                                <div class="flex-1 min-w-0">
                                    <p class="text-sm text-[var(--wz-on-surface)]"><?php echo esc_html( $tx['description'] ?? '' ); ?></p>
                                    <p class="text-[10px] text-[var(--wz-outline)]"><?php echo esc_html( $tx['date'] ?? '' ); ?></p>
                                </div>
                                <span class="text-sm font-bold <?php echo $is_positive ? 'text-[var(--wz-accent)]' : 'text-[var(--wz-error)]'; ?>"><?php echo $is_positive ? '+' : '-'; ?><?php echo esc_html( wz_format_price( (float) ( $tx['amount'] ?? 0 ) ) ); ?></span>
                            </div>
                        <?php endforeach; ?>
                    </div>
                <?php endif; ?>
            </div>
        </div>
    </div>

</div>
</div>

<?php get_footer(); ?>
```
