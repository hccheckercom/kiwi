## Bad
```ts
export async function POST(req: Request) {
  const secret = req.headers.get('x-sepay-token') ?? ''
  const expected = process.env.SEPAY_WEBHOOK_SECRET ?? ''

  // String === comparison leaks timing info — attacker can brute-force byte by byte
  if (secret !== expected) {
    return Response.json({ error: 'Unauthorized' }, { status: 401 })
  }
}
```

## Good (pattern from webhook.helpers.ts)
```ts
import { timingSafeEqual } from 'crypto'

export function timingSafeCompare(a: string, b: string): boolean {
  if (a.length !== b.length) return false
  return timingSafeEqual(Buffer.from(a), Buffer.from(b))
}

export async function POST(req: Request) {
  const expectedSecret = process.env.SEPAY_WEBHOOK_SECRET ?? ''
  if (!expectedSecret) {
    return Response.json({ error: 'Service misconfigured' }, { status: 503 })
  }

  const secret = req.headers.get('x-sepay-token') ?? ''
  if (!timingSafeCompare(secret, expectedSecret)) {
    return Response.json({ error: 'Unauthorized' }, { status: 401 })
  }
}
```

## Also: guard against unconfigured secret
```ts
// If SEPAY_WEBHOOK_SECRET is empty, reject ALL requests (503)
// Don't silently accept everything when secret is missing
if (!expectedSecret) {
  console.error('[Webhook] SEPAY_WEBHOOK_SECRET not configured')
  return Response.json({ error: 'Service misconfigured' }, { status: 503 })
}
```

## Why
Real pattern from webhook.handler.ts. String === comparison short-circuits on first mismatch — response time reveals how many bytes matched. With enough requests, attacker can reconstruct the secret byte by byte. timingSafeEqual always takes constant time regardless of where strings differ. Also: empty secret must reject (503), not accept all.
