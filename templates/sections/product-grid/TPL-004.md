Grid sản phẩm nổi bật. Header: eyebrow nhỏ + heading serif italic + "Tất cả" link bên phải. Grid cards: ảnh vuông với discount badge góc trái + quick-add button góc phải dưới, info bên dưới (brand, title 2-line clamp, rating, price). Hover: shadow + translateY.

## Code — PHP

```php
<?php
$args = array(
    'post_type'      => 'wz_product',
    'posts_per_page' => 8,
    'post_status'    => 'publish',
    'orderby'        => 'date',
    'order'          => 'DESC',
);

if ( ! post_type_exists( 'wz_product' ) ) {
    $args['post_type'] = 'product';
}

$products = new WP_Query( $args );

if ( ! $products->have_posts() ) {
    return;
}

$shop_url = function_exists( 'wz_get_shop_url' ) ? wz_get_shop_url() : home_url( '/shop/' );
?>

<section style="padding:40px 24px;max-width:1280px;margin:0 auto;">
    <div class="haven-section-header--row">
        <div>
            <p class="haven-eyebrow"><?php esc_html_e( 'Tuyển chọn', 'wezone-haven' ); ?></p>
            <h2 class="haven-heading"><?php esc_html_e( 'Sản phẩm nổi bật', 'wezone-haven' ); ?></h2>
        </div>
        <a href="<?php echo esc_url( $shop_url ); ?>" class="haven-view-all">
            <?php esc_html_e( 'Tất cả', 'wezone-haven' ); ?>
        </a>
    </div>

    <div style="display:grid;grid-template-columns:repeat(2,1fr);gap:16px;" class="haven-product-grid">
        <?php while ( $products->have_posts() ) : $products->the_post();
            $price     = function_exists( 'wz_get_product_price' ) ? wz_get_product_price( get_the_ID() ) : get_post_meta( get_the_ID(), '_price', true );
            $old_price = function_exists( 'wz_get_product_regular_price' ) ? wz_get_product_regular_price( get_the_ID() ) : get_post_meta( get_the_ID(), '_regular_price', true );
            $brand     = get_post_meta( get_the_ID(), '_brand', true );
            $rating    = get_post_meta( get_the_ID(), '_wz_average_rating', true );
            $reviews   = get_post_meta( get_the_ID(), '_wz_review_count', true );
        ?>
            <a href="<?php the_permalink(); ?>"
                style="display:flex;flex-direction:column;background:#fff;border:1px solid var(--wz-border,#cdc6b7);border-radius:8px;overflow:hidden;text-decoration:none;transition:box-shadow 0.25s,transform 0.25s;"
                onmouseover="this.style.boxShadow='0 8px 24px rgba(81,70,31,0.1)';this.style.transform='translateY(-2px)'"
                onmouseout="this.style.boxShadow='none';this.style.transform='none'">

                <div style="position:relative;aspect-ratio:1/1;overflow:hidden;background:var(--wz-surface,#F5F0E8);">
                    <?php if ( has_post_thumbnail() ) : ?>
                        <?php the_post_thumbnail( 'wezone-card', array( 'style' => 'width:100%;height:100%;object-fit:cover;' ) ); ?>
                    <?php else : ?>
                        <div style="width:100%;height:100%;display:flex;align-items:center;justify-content:center;">
                            <span class="material-symbols-outlined" aria-hidden="true" style="font-size:48px;color:var(--wz-border);">image</span>
                        </div>
                    <?php endif; ?>

                    <?php if ( $old_price && $price && (float) $old_price > (float) $price ) :
                        $discount = round( ( 1 - (float) $price / (float) $old_price ) * 100 );
                    ?>
                        <span style="position:absolute;top:10px;left:10px;background:var(--wz-text,#3D2B1F);color:#fff;font-size:10px;font-weight:600;padding:3px 8px;letter-spacing:0.05em;">
                            -<?php echo esc_html( (string) $discount ); ?>%
                        </span>
                    <?php endif; ?>
                </div>

                <div style="padding:14px;">
                    <?php if ( $brand ) : ?>
                        <div style="font-size:10px;font-weight:600;letter-spacing:0.1em;text-transform:uppercase;color:var(--wz-text-muted);margin-bottom:4px;">
                            <?php echo esc_html( $brand ); ?>
                        </div>
                    <?php endif; ?>

                    <div style="font-size:14px;font-weight:500;color:var(--wz-text);line-height:1.4;margin-bottom:8px;display:-webkit-box;-webkit-line-clamp:2;-webkit-box-orient:vertical;overflow:hidden;">
                        <?php the_title(); ?>
                    </div>

                    <?php if ( $rating ) : ?>
                        <div style="display:flex;align-items:center;gap:4px;margin-bottom:8px;">
                            <span style="font-size:12px;color:var(--wz-primary);">★ <?php echo esc_html( number_format( (float) $rating, 1 ) ); ?></span>
                            <?php if ( $reviews ) : ?>
                                <span style="font-size:11px;color:var(--wz-text-muted);">(<?php echo esc_html( (string) $reviews ); ?>)</span>
                            <?php endif; ?>
                        </div>
                    <?php endif; ?>

                    <div style="display:flex;align-items:baseline;gap:8px;">
                        <?php if ( $price ) : ?>
                            <span style="font-size:15px;font-weight:700;color:var(--wz-text);">
                                <?php echo function_exists( 'wz_format_price' ) ? esc_html( wz_format_price( (float) $price ) ) : esc_html( number_format( (float) $price, 0, ',', '.' ) ) . 'đ'; ?>
                            </span>
                        <?php endif; ?>
                        <?php if ( $old_price && $price && (float) $old_price > (float) $price ) : ?>
                            <span style="font-size:12px;color:var(--wz-text-muted);text-decoration:line-through;">
                                <?php echo function_exists( 'wz_format_price' ) ? esc_html( wz_format_price( (float) $old_price ) ) : esc_html( number_format( (float) $old_price, 0, ',', '.' ) ) . 'đ'; ?>
                            </span>
                        <?php endif; ?>
                    </div>
                </div>
            </a>
        <?php endwhile; wp_reset_postdata(); ?>
    </div>
</section>

<style>
@media (min-width: 768px)  { .haven-product-grid { grid-template-columns: repeat(3, 1fr) !important; } }
@media (min-width: 1024px) { .haven-product-grid { grid-template-columns: repeat(4, 1fr) !important; } }
</style>
```

## Usage Notes

- Dùng `wz_product` post type, fallback `product` nếu chưa register
- `wz_get_product_price()` / `wz_get_product_regular_price()` — fallback post meta
- `wz_format_price()` — fallback number_format
- Placeholder icon khi chưa có thumbnail
- Section heading dùng Haven heading system (eyebrow + heading + view-all)
