## Bad
```php
// Frontend: useWebSocket.ts
useEffect(() => {
  const ws = new WebSocket('ws://localhost:8000/ws')
  
  ws.onopen = () => {
    console.log('WebSocket connected')
    setConnected(true)
  }
  
  ws.onerror = (event) => {
    console.error('WebSocket error:', event)
  }
  
  ws.onclose = () => {
    console.log('WebSocket disconnected')
    setConnected(false)
  }
}, [])

// Backend: api.py
@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    await manager.connect(websocket)
    try:
        while True:
            data = await websocket.receive_text()
            # Process data...
    except WebSocketDisconnect:
        manager.disconnect(websocket)
```

## Good
```php
// Frontend: useWebSocket.ts
useEffect(() => {
  const ws = new WebSocket('ws://localhost:8000/ws')
  
  ws.onmessage = (event) => {
    const msg = JSON.parse(event.data)
    if (msg.type === 'connected') {
      console.log('WebSocket connected:', msg.status)
      setConnected(true)
    }
  }
  
  ws.onerror = (event) => {
    console.error('WebSocket error:', event)
  }
  
  ws.onclose = () => {
    console.log('WebSocket disconnected')
    setConnected(false)
  }
}, [])

// Backend: api.py
@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    try:
        await manager.connect(websocket)
        await websocket.send_json({"type": "connected", "status": "ok"})
        
        while True:
            data = await websocket.receive_text()
            # Process data...
    except WebSocketDisconnect:
        manager.disconnect(websocket)
    except Exception as e:
        logger.error(f"WebSocket error: {e}")
        try:
            await websocket.close(code=1011, reason=str(e))
        except:
            pass
        manager.disconnect(websocket)
```

## Why
When WebSocket server calls accept() but doesn't send initial message, client sees "WebSocket is closed before the connection is established" error. This happens because client's onopen handler waits for first message to confirm connection succeeded. Without it, connection appears to hang then timeout. Solution: send confirmation message immediately after accept(), and handle errors properly with try/except around entire handler.
