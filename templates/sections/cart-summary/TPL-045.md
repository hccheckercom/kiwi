if ( ! defined( 'ABSPATH' ) ) {
	exit; }
$cart_data  = function_exists( 'wezone_trunganh_v2_read_cart_from_db' ) ? wezone_trunganh_v2_read_cart_from_db() : array();
$cart_items = $cart_data['items'] ?? array();
$subtotal   = 0;
foreach ( $cart_items as $item ) {
	$subtotal += ( $item['price'] ?? 0 ) * ( $item['qty'] ?? $item['quantity'] ?? 1 );
}
$ship_fee = (int) wz_config( 'shipping.fees.default', 0 );
$total    = $subtotal + $ship_fee;
$count    = count( $cart_items );
$phone    = wz_config( 'phone', '' );
?>
<div class="flex flex-col gap-lg">
	<div class="bg-white p-lg rounded-xl shadow-md border border-surface-container-high">
		<h2 class="font-h3 text-h3 mb-md pb-sm" style="border-bottom:1px solid var(--wz-outline-variant)">Tóm tắt đơn hàng</h2>
		<div class="space-y-sm mb-md">
			<div class="flex justify-between items-center">
				<span class="text-sm text-on-surface-variant">Tạm tính (<?php echo $count; ?> sản phẩm)</span>
				<span class="font-bold text-sm text-on-surface" id="wz-cart-subtotal"><?php echo esc_html( wz_format_price( $subtotal ) ); ?></span>
			</div>
			<div class="flex justify-between items-center">
				<span class="text-sm text-on-surface-variant">Phí vận chuyển</span>
				<span class="text-sm <?php echo $ship_fee === 0 ? 'text-success font-bold' : 'text-on-surface'; ?>" id="wz-cart-shipping">
					<?php echo $ship_fee === 0 ? 'Miễn phí' : esc_html( wz_format_price( $ship_fee ) ); ?>
				</span>
			</div>
			<div class="flex justify-between items-center hidden" id="cart-discount-row">
				<span class="text-sm text-on-surface-variant">Giảm giá voucher</span>
				<span class="text-sm font-bold text-success" id="cart-discount-amount">-0đ</span>
			</div>
		</div>
		<div class="pt-md mb-lg" style="border-top:1px solid var(--wz-outline-variant)">
			<div class="flex justify-between items-center">
				<span class="font-bold text-on-surface uppercase tracking-wide text-sm">Tổng cộng</span>
				<div class="text-right">
					<span class="font-h2 text-h2 text-tertiary block" id="cart-total"><?php echo esc_html( wz_format_price( $total ) ); ?></span>
					<span class="text-[11px] text-outline">(Đã bao gồm VAT)</span>
				</div>
			</div>
		</div>
		<a href="<?php echo esc_url( function_exists( 'wz_get_checkout_url' ) ? wz_get_checkout_url() : home_url( '/checkout/' ) ); ?>"
			class="w-full bg-accent text-white py-md rounded-lg font-h3 text-h3 hover:brightness-110 shadow-lg active:scale-[0.98] transition-all flex items-center justify-center gap-sm">
			<span>TIẾN HÀNH THANH TOÁN</span>
			<?php wz_icon( 'arrow_forward', [ 'size' => 'sm' ] ); ?>
		</a>
		<p class="mt-md flex items-center gap-xs justify-center text-xs text-outline">
			<?php wz_icon( 'verified_user', [ 'size' => 'xs' ] ); ?>
			Thanh toán an toàn &amp; bảo mật
		</p>
		<div class="mt-sm pt-sm" style="border-top:1px solid var(--wz-outline-variant)">
			<p class="text-[10px] text-outline text-center uppercase tracking-wider mb-xs">Chấp nhận thanh toán</p>
			<div class="flex items-center justify-center gap-sm flex-wrap">
				<span class="px-sm py-xs rounded border text-xs font-bold text-on-surface-variant" style="border-color:var(--wz-outline-variant)">Visa</span>

				<span class="px-sm py-xs rounded border text-xs font-bold text-primary" style="border-color:var(--wz-primary)">COD</span>
			</div>
		</div>
	</div>
	<?php if ( $phone ) : ?>
	<div class="bg-white p-md rounded-xl shadow-md flex flex-col gap-xs">
		<p class="text-xs text-on-surface-variant">Bạn cần hỗ trợ?</p>
		<a href="tel:<?php echo esc_attr( preg_replace( '/\D/', '', $phone ) ); ?>"
			class="flex items-center gap-sm text-primary font-bold hover:underline">
			<?php wz_icon( 'call', [ 'size' => 'sm', 'color' => 'primary' ] ); ?>
			<span class="text-sm"><?php echo esc_html( $phone ); ?></span>
		</a>
		<p class="text-xs text-outline">(8:00 - 21:00)</p>
	</div>
	<?php endif; ?>
