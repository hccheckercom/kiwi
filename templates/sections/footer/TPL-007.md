## Preview

Footer nền tối (accent color). 4 cột: Brand (logo text + tagline + social) | Policies (list links) | Contact (icon + text) | Newsletter (form email + button). Dưới cùng: copyright + powered by. Mobile: 1 cột stack.

## Files

| File | Role |
|------|------|
| footer.php | Template chính + inline JS menu |
| template-parts/footer/mobile-nav.php | Mobile bottom nav (optional) |

## Code — PHP

```php
<?php
if ( ! function_exists( 'wz_config' ) ) {
    echo '</body></html>';
    return;
}

$shop_name  = wz_config( 'name', get_bloginfo( 'name' ) );
$tagline    = wz_config( 'tagline', '' );
$phone      = wz_config( 'phone', '' );
$email      = wz_config( 'email', '' );
$address    = wz_config( 'address', '' );
$copyright  = wz_config( 'footer.copyright', '© ' . gmdate( 'Y' ) . ' ' . $shop_name );
$powered_by = wz_config( 'footer.powered_by', true );
$zalo_url   = wz_config( 'zalo_url', '' );
$facebook   = wz_config( 'facebook_url', '' );

$policy_links = array(
    'chinh-sach-van-chuyen' => __( 'Chính sách vận chuyển', 'wezone-haven' ),
    'chinh-sach-doi-tra'    => __( 'Chính sách đổi trả', 'wezone-haven' ),
    'chinh-sach-bao-mat'    => __( 'Chính sách bảo mật', 'wezone-haven' ),
    'dieu-khoan-su-dung'    => __( 'Điều khoản sử dụng', 'wezone-haven' ),
    'huong-dan-mua-hang'    => __( 'Hướng dẫn mua hàng', 'wezone-haven' ),
    'lien-he'               => __( 'Liên hệ', 'wezone-haven' ),
);
?>

<footer class="wz-footer">
    <div style="max-width:1280px;margin:0 auto;">
        <div class="wz-footer__grid">

            <!-- Brand -->
            <div>
                <div style="font-family:var(--wz-font-heading);font-size:22px;font-weight:600;color:#fff;letter-spacing:0.08em;margin-bottom:12px;">
                    <?php echo esc_html( strtoupper( $shop_name ) ); ?>
                </div>
                <?php if ( $tagline ) : ?>
                    <p style="font-size:13px;color:rgba(255,255,255,0.65);line-height:1.6;margin-bottom:16px;"><?php echo esc_html( $tagline ); ?></p>
                <?php endif; ?>
                <div class="wz-footer__social">
                    <?php if ( $facebook ) : ?>
                        <a href="<?php echo esc_url( $facebook ); ?>" target="_blank" rel="noopener noreferrer" aria-label="Facebook">Facebook</a>
                    <?php endif; ?>
                    <?php if ( $zalo_url ) : ?>
                        <a href="<?php echo esc_url( $zalo_url ); ?>" target="_blank" rel="noopener noreferrer" aria-label="Zalo">Zalo</a>
                    <?php endif; ?>
                </div>
            </div>

            <!-- Policies -->
            <div>
                <h4 class="wz-footer__heading"><?php esc_html_e( 'Chính sách', 'wezone-haven' ); ?></h4>
                <ul class="wz-footer__list">
                    <?php foreach ( $policy_links as $slug => $label ) : ?>
                        <li><a href="<?php echo esc_url( home_url( '/' . $slug . '/' ) ); ?>"><?php echo esc_html( $label ); ?></a></li>
                    <?php endforeach; ?>
                </ul>
            </div>

            <!-- Contact -->
            <div>
                <h4 class="wz-footer__heading"><?php esc_html_e( 'Liên hệ', 'wezone-haven' ); ?></h4>
                <div class="wz-footer__contact">
                    <?php if ( $phone ) : ?>
                        <div class="wz-footer__contact-item">
                            <span class="material-symbols-outlined" aria-hidden="true">call</span>
                            <a href="tel:<?php echo esc_attr( preg_replace( '/\s+/', '', $phone ) ); ?>"><?php echo esc_html( $phone ); ?></a>
                        </div>
                    <?php endif; ?>
                    <?php if ( $email ) : ?>
                        <div class="wz-footer__contact-item">
                            <span class="material-symbols-outlined" aria-hidden="true">mail</span>
                            <a href="mailto:<?php echo esc_attr( $email ); ?>"><?php echo esc_html( $email ); ?></a>
                        </div>
                    <?php endif; ?>
                    <?php if ( $address ) : ?>
                        <div class="wz-footer__contact-item">
                            <span class="material-symbols-outlined" aria-hidden="true">location_on</span>
                            <span><?php echo esc_html( $address ); ?></span>
                        </div>
                    <?php endif; ?>
                </div>
            </div>

            <!-- Newsletter -->
            <div>
                <h4 class="wz-footer__heading"><?php esc_html_e( 'Đăng ký nhận tin', 'wezone-haven' ); ?></h4>
                <p style="font-size:13px;color:rgba(255,255,255,0.65);margin-bottom:12px;"><?php esc_html_e( 'Nhận ưu đãi sớm và xu hướng mới.', 'wezone-haven' ); ?></p>
                <form style="display:flex;gap:8px;">
                    <label for="wz-newsletter-email" class="screen-reader-text"><?php esc_html_e( 'Email', 'wezone-haven' ); ?></label>
                    <input id="wz-newsletter-email" type="email" placeholder="<?php esc_attr_e( 'Email của bạn', 'wezone-haven' ); ?>"
                        style="flex:1;padding:10px 14px;border:1px solid rgba(255,255,255,0.2);border-radius:var(--wz-radius,4px);background:rgba(255,255,255,0.08);color:#fff;font-size:13px;outline:none;" />
                    <button type="submit" style="padding:10px 16px;background:var(--wz-primary,#8B7355);color:#fff;border:none;border-radius:var(--wz-radius,4px);font-size:13px;font-weight:500;cursor:pointer;">
                        <?php esc_html_e( 'Gửi', 'wezone-haven' ); ?>
                    </button>
                </form>
            </div>

        </div>

        <div class="wz-footer__bottom">
            <p><?php echo esc_html( $copyright ); ?></p>
            <?php if ( $powered_by ) : ?>
                <p class="wz-footer__powered"><?php esc_html_e( 'Powered by', 'wezone-haven' ); ?> <a href="https://wezone.vn">WeZone</a></p>
            <?php endif; ?>
        </div>
    </div>
</footer>

<?php wp_footer(); ?>
</body>
</html>
```

## Usage Notes

- Tất cả content dynamic từ `wz_config()` — chỉ cần cập nhật `store-config.php`
- Policy links cần tạo pages tương ứng trong WordPress
- Newsletter form cần backend handler (hoặc integrate Mailchimp/Zalo OA)
- Mobile nav (bottom fixed bar) là file riêng, optional
