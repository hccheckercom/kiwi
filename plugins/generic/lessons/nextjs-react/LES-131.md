## Bad
```ts
import { S3Client } from '@aws-sdk/client-s3'

// Module-level init — runs at import time
// In production: env vars may not be loaded yet → empty credentials
const s3 = new S3Client({
  region: process.env.S3_REGION!,
  credentials: {
    accessKeyId: process.env.S3_ACCESS_KEY!,
    secretAccessKey: process.env.S3_SECRET_KEY!,
  },
})

export function uploadFile(key: string, body: Buffer) {
  return s3.send(new PutObjectCommand({ Bucket: process.env.S3_BUCKET!, Key: key, Body: body }))
}
```

## Good (lazy init)
```ts
import { S3Client } from '@aws-sdk/client-s3'

let _client: S3Client | null = null

function getS3Client() {
  if (!_client) {
    _client = new S3Client({
      region: process.env.S3_REGION!,
      credentials: {
        accessKeyId: process.env.S3_ACCESS_KEY!,
        secretAccessKey: process.env.S3_SECRET_KEY!,
      },
    })
  }
  return _client
}

export function uploadFile(key: string, body: Buffer) {
  return getS3Client().send(new PutObjectCommand({ Bucket: process.env.S3_BUCKET!, Key: key, Body: body }))
}
```

## Why
Real bug: commit a2bfe7e. Module-level client initialization reads process.env at import time. In Next.js standalone production builds, env vars may not be available during module evaluation (especially with PM2 ecosystem.config.js). Lazy-init ensures credentials are read at call time.
