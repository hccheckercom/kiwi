## Bad (original webstore-vn middleware)
```ts
// matcher: ['/((?!_next|api).*)'] — matches ALL pages
export async function middleware(req: NextRequest) {
  const { data: { user } } = await supabase.auth.getUser()
  // DB query on EVERY page including public (blog, pricing, template...)
  const { data: profile } = await supabase.from('profiles').select('role').eq('id', user.id).single()
}
// Result: +100-300ms latency on 80-90% of traffic that doesn't need auth
```

## Good (current webstore-vn middleware after FIX M-04)
```ts
export const config = {
  matcher: ['/dashboard/:path*', '/admin/:path*', '/login', '/auth/callback'],
}

export async function middleware(req: NextRequest) {
  const { data: { user } } = await supabase.auth.getUser()
  // JWT fast path — read role from app_metadata (no DB query)
  const jwtRole = user?.app_metadata?.role
  if (jwtRole === 'admin') return // 0ms DB overhead

  // Fallback DB query ONLY for legacy tokens without app_metadata.role
  if (!jwtRole && decodedPathname.startsWith('/admin')) {
    const { data: profile } = await supabase.from('profiles').select('role')...
  }
}
```

## Why
Real fix: FIX M-04 in middleware.ts. Broad matcher + DB query = 100-300ms added to every page load. Fix: (1) narrow matcher to auth-required routes only, (2) read role from JWT app_metadata first (zero DB queries for 99% of requests), (3) fallback DB query only for legacy tokens.
