## Bad
```ts
// Cron: runs every 5 minutes
export async function GET() {
  const { data: pendingOrders } = await supabase
    .from('orders').select('*').eq('status', 'pending').lt('created_at', fiveMinutesAgo)
  for (const order of pendingOrders) {
    await cancelOrder(order.id) // May run on same orders if previous run is still processing
  }
}
```

## Good
```ts
export async function GET() {
  // Advisory lock prevents concurrent execution
  const { data: locked } = await supabase.rpc('try_cron_lock', { lock_name: 'cancel_stale_orders' })
  if (!locked) return Response.json({ skipped: 'already running' })

  try {
    const { data: pendingOrders } = await supabase
      .from('orders').select('*').eq('status', 'pending').lt('created_at', fiveMinutesAgo)
    for (const order of pendingOrders) {
      await cancelOrder(order.id)
    }
  } finally {
    await supabase.rpc('release_cron_lock', { lock_name: 'cancel_stale_orders' })
  }
}
```

## Why
Cron jobs can overlap if the previous run takes longer than the interval. Without locking, the same records are processed by multiple invocations — causing duplicate emails, double refunds, or inconsistent state.
