## Pattern

WordPress nonce has 24h lifetime (2 ticks × 12h). Admin SPA loads nonce once via `wp_localize_script()`. If admin keeps tab open > 12h (overnight), all API calls silently fail with 403. No refresh mechanism → admin thinks system is broken.

## Scan

```yaml
pattern: 'wp_create_nonce|wp_localize_script.*nonce'
scope: "**/AdminUI/**/*.php|**/admin*/**/*.php"
exclude_line: "heartbeat|refresh_nonce|nonce_refresh"
message: "Admin nonce localized once — expires after 12-24h without refresh"
```

## Bug

```php
// PHP: nonce set 1 lần khi page load
wp_localize_script( 'wezone-admin-app', 'wzAdmin', array(
    'nonce' => wp_create_nonce( 'wp_rest' ),
    'apiUrl' => rest_url( 'wezone/v1/' ),
));

// JS: dùng nonce cũ mãi mãi
const wzApi = {
  get: (endpoint) => fetch(wzAdmin.apiUrl + endpoint, {
    headers: { 'X-WP-Nonce': wzAdmin.nonce }
  })
};
// Sau 12h+ → 403 Forbidden, admin không hiểu tại sao
```

## Fix

```php
// PHP: thêm endpoint refresh nonce
register_rest_route( 'wezone/v1', '/nonce', array(
    'methods'             => 'GET',
    'callback'            => function() {
        return new WP_REST_Response( array(
            'nonce' => wp_create_nonce( 'wp_rest' ),
        ) );
    },
    'permission_callback' => fn() => current_user_can( 'manage_options' ),
) );
```

```tsx
// JS: auto-refresh nonce khi gặp 403 hoặc mỗi 10h
const refreshNonce = async () => {
  const res = await fetch(wzAdmin.apiUrl + 'nonce', {
    credentials: 'same-origin',
  });
  if (res.ok) {
    const data = await res.json();
    wzAdmin.nonce = data.nonce;
  }
};

// Intercept 403 → refresh + retry
const wzFetch = async (url, options = {}) => {
  let res = await fetch(url, { ...options, headers: { ...options.headers, 'X-WP-Nonce': wzAdmin.nonce } });
  if (res.status === 403) {
    await refreshNonce();
    res = await fetch(url, { ...options, headers: { ...options.headers, 'X-WP-Nonce': wzAdmin.nonce } });
  }
  return res;
};
```

## Áp dụng

Admin SPA PHẢI có nonce refresh mechanism. Hoặc: (1) intercept 403 → refresh + retry, hoặc (2) dùng WordPress Heartbeat API để refresh mỗi 15 phút.
