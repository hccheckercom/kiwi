                        <?php endforeach; ?>
                    </div>
                <?php endif; ?>

            <a href="<?php echo esc_url( wz_get_shop_url() ); ?>" class="inline-flex items-center gap-2 px-5 py-2.5 bg-[var(--wz-primary)] text-white rounded-[var(--wz-radius)] text-sm font-semibold hover:brightness-110 transition">Mua sắm để tích điểm</a>
        </div>
    </div>

</div>
</div>

<?php get_footer(); ?>
