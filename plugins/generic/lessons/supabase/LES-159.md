## Bad
```ts
export async function validateApiKey(apiKey: string): Promise<PlatformStore | null> {
  // No rate limit — attacker can try millions of keys
  const { data } = await supabase.from('platform_stores')
    .select().eq('api_key', apiKey).eq('status', 'active').single()
  return data
}
```

## Good (add rate limit + early rejection)
```ts
import { checkRateLimit } from '@/lib/rate-limit'

export async function validateApiKey(apiKey: string): Promise<PlatformStore | null> {
  // Early rejection — minimum length check (no DB query for garbage)
  if (!apiKey || apiKey.length < 10) return null

  // Rate limit per IP — prevent brute-force
  // (Applied in withPlatformKey middleware, not here)

  try {
    const { data, error } = await supabase.from('platform_stores')
      .select().eq('api_key', apiKey).eq('status', 'active').single()
    if (error || !data) return null
    return data as PlatformStore
  } catch {
    return null
  }
}
```

## Rate limit at middleware level
```ts
// In withPlatformKey or route handler:
const ip = req.headers.get('x-forwarded-for') ?? 'unknown'
if (!checkRateLimit(`platform:${ip}`, 60, 60_000)) {
  return apiError('RATE_LIMITED', 'Too many requests', 429)
}
```

## Why
Platform API keys authenticate store owners. Without rate limiting, an attacker can enumerate valid keys by trying random strings. The existing checkRateLimit (lib/rate-limit.ts) provides in-process fixed-window limiting — apply it to the platform auth path. Also: minimum length check (apiKey.length < 10) rejects garbage without hitting DB.
