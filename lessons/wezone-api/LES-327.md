## Pattern

Admin imports products/customers via CSV upload. Backend reads CSV and inserts directly without: (1) validating headers match expected columns, (2) validating each row's data types, (3) checking for duplicates, (4) limiting file size. Result: wrong column order → prices in name field, empty rows → NULL violations, 500MB file → PHP timeout.

## Bug

```php
// SAI — no header check, no row validation, no size limit
public function import_products( WP_REST_Request $request ): WP_REST_Response {
    $file = $request->get_file_params()['csv'];
    $handle = fopen( $file['tmp_name'], 'r' );

    fgetcsv( $handle ); // Skip header — doesn't even check what it is!

    while ( ( $row = fgetcsv( $handle ) ) !== false ) {
        $wpdb->insert( $table, array(
            'name'  => $row[0],  // Có thể là price nếu column order sai!
            'price' => $row[1],  // Có thể là "abc" → 0
            'sku'   => $row[2],  // Có thể duplicate
        ) );
    }

    fclose( $handle );
    return new WP_REST_Response( array( 'imported' => 'done' ) );
}
```

## Fix

```php
private const REQUIRED_HEADERS = array( 'name', 'price', 'sku', 'category', 'stock_qty' );
private const MAX_CSV_SIZE     = 5 * 1024 * 1024; // 5MB
private const MAX_ROWS         = 5000;

public function import_products( WP_REST_Request $request ): WP_REST_Response {
    global $wpdb;
    $file = $request->get_file_params()['csv'] ?? null;

    if ( ! $file || UPLOAD_ERR_OK !== $file['error'] ) {
        return new WP_REST_Response( array( 'error' => 'Upload failed' ), 400 );
    }

    // 1. Size limit
    if ( $file['size'] > self::MAX_CSV_SIZE ) {
        return new WP_REST_Response( array( 'error' => 'File quá lớn (max 5MB)' ), 422 );
    }

    $handle = fopen( $file['tmp_name'], 'r' );
    if ( ! $handle ) {
        return new WP_REST_Response( array( 'error' => 'Cannot read file' ), 500 );
    }

    // 2. Validate headers
    $headers = array_map( 'strtolower', array_map( 'trim', fgetcsv( $handle ) ) );
    $missing = array_diff( self::REQUIRED_HEADERS, $headers );
    if ( $missing ) {
        fclose( $handle );
        return new WP_REST_Response(
            array( 'error' => 'Thiếu cột: ' . implode( ', ', $missing ) ),
            422
        );
    }

    $col_map = array_flip( $headers );

    // 3. Validate + collect rows
    $rows    = array();
    $errors  = array();
    $line    = 1;

    while ( ( $row = fgetcsv( $handle ) ) !== false ) {
        $line++;
        if ( count( $rows ) >= self::MAX_ROWS ) {
            $errors[] = "Dòng {$line}: Đã đạt giới hạn " . self::MAX_ROWS . ' sản phẩm';
            break;
        }

        $name  = trim( $row[ $col_map['name'] ] ?? '' );
        $price = $row[ $col_map['price'] ] ?? '';
        $sku   = trim( $row[ $col_map['sku'] ] ?? '' );

        // Validate each row
        if ( empty( $name ) ) {
            $errors[] = "Dòng {$line}: Tên sản phẩm trống";
            continue;
        }
        if ( ! is_numeric( $price ) || (float) $price < 0 ) {
            $errors[] = "Dòng {$line}: Giá không hợp lệ '{$price}'";
            continue;
        }
        if ( empty( $sku ) ) {
            $errors[] = "Dòng {$line}: SKU trống";
            continue;
        }

        $rows[] = array(
            'name'      => sanitize_text_field( $name ),
            'price'     => round( (float) $price, 0 ),
            'sku'       => sanitize_text_field( $sku ),
            'stock_qty' => max( 0, (int) ( $row[ $col_map['stock_qty'] ] ?? 0 ) ),
        );
    }
    fclose( $handle );

    if ( empty( $rows ) ) {
        return new WP_REST_Response( array( 'error' => 'Không có dòng hợp lệ', 'details' => $errors ), 422 );
    }

    // 4. Check SKU duplicates against DB
    $skus       = array_column( $rows, 'sku' );
    $placeholders = implode( ',', array_fill( 0, count( $skus ), '%s' ) );
    $existing   = $wpdb->get_col( $wpdb->prepare(
        "SELECT sku FROM {$wpdb->prefix}wz_products WHERE sku IN ({$placeholders})",
        ...$skus
    ) );

    if ( $existing ) {
        $errors[] = 'SKU đã tồn tại: ' . implode( ', ', array_slice( $existing, 0, 10 ) );
        $rows = array_filter( $rows, fn( $r ) => ! in_array( $r['sku'], $existing, true ) );
    }

    // 5. Bulk insert with transaction
    if ( ! empty( $rows ) ) {
        wz_bulk_insert( $wpdb->prefix . 'wz_products', $rows, array( '%s', '%f', '%s', '%d' ) );
    }

    return new WP_REST_Response( array(
        'imported' => count( $rows ),
        'skipped'  => count( $errors ),
        'errors'   => array_slice( $errors, 0, 20 ),
    ) );
}
```

## Áp dụng

Admin CSV import PHẢI: (1) validate file size, (2) validate headers match expected, (3) validate each row data type + required fields, (4) check duplicates (SKU/email), (5) limit max rows, (6) return detailed error report per row. KHÔNG insert trực tiếp từ raw CSV.
