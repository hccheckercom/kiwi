## Bad
```ts
export async function handleVnpayGet(req: Request) {
  const { searchParams } = new URL(req.url)
  const responseCode = searchParams.get('vnp_ResponseCode')
  const orderCode = searchParams.get('vnp_TxnRef')

  // No signature verification — attacker crafts URL with responseCode=00
  if (responseCode === '00') {
    return Response.redirect(`/don-hang/${orderCode}?paid=1`)
  }
}
```

## Good (pattern from callback.handlers.ts)
```ts
export async function handleVnpayGet(req: Request) {
  const { searchParams } = new URL(req.url)
  const responseCode = searchParams.get('vnp_ResponseCode')
  const orderCode = searchParams.get('vnp_TxnRef')

  // Verify HMAC SHA512 signature BEFORE trusting any params
  const HASH_SECRET = process.env.VNPAY_HASH_SECRET
  if (HASH_SECRET) {
    const receivedHash = searchParams.get('vnp_SecureHash')
    if (receivedHash && !verifyVnpaySearchParams(searchParams, HASH_SECRET)) {
      return Response.redirect(`/don-hang/${orderCode}?paid=0&error=INVALID_SIGNATURE`)
    }
  }

  if (responseCode === '00') {
    return Response.redirect(`/don-hang/${orderCode}?paid=1`)
  }
}
```

## IPN handler (POST) — same pattern
```ts
export async function handleVnpayPost(req: Request) {
  if (!params.vnp_SecureHash) {
    return Response.json({ RspCode: '97', Message: 'Missing signature' })
  }
  if (!verifyVnpaySignature(params, HASH_SECRET)) {
    return Response.json({ RspCode: '97', Message: 'Checksum failed' })
  }
  // Only process payment AFTER signature verified
}
```

## Why
Real pattern from VNPay integration. The return URL (GET) is user-facing — anyone can craft a URL with vnp_ResponseCode=00. Without HMAC verification, the UI shows "payment successful" for unpaid orders. The IPN (POST) is server-to-server but still needs verification to prevent replay attacks.
