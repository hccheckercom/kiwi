```php
<!-- notifications.php -->
<?php
declare(strict_types=1);

if ( ! defined( 'ABSPATH' ) ) { exit; }

if ( ! is_user_logged_in() ) {
    wp_safe_redirect( home_url( '/my-account/login/' ) );
    exit;
}

get_header();

$user_id      = get_current_user_id();
$unread_count = (int) get_user_meta( $user_id, '_wz_unread_notifications', true );
$notifications = function_exists( 'wz_get_user_notifications' ) ? wz_get_user_notifications( $user_id, [ 'limit' => 20 ] ) : [];

$type_icons = [
    'order'     => [ 'icon' => 'local_shipping', 'bg' => 'bg-[var(--wz-primary)]/10', 'color' => 'text-[var(--wz-primary)]' ],
    'promo'     => [ 'icon' => 'sell',           'bg' => 'bg-[var(--wz-accent)]/10',  'color' => 'text-[var(--wz-accent)]' ],
    'system'    => [ 'icon' => 'info',           'bg' => 'bg-[var(--wz-outline)]/10', 'color' => 'text-[var(--wz-outline)]' ],
];

set_query_var( 'wz_tpl_account_page', 'notifications' );
?>

<div class="bg-[var(--wz-surface)] min-h-screen">
<div class="max-w-[1280px] mx-auto px-5 md:px-[20px] py-4">

    <nav aria-label="Breadcrumb" class="pb-4">
        <ol class="flex items-center gap-1 text-xs md:text-sm text-[var(--wz-outline)]">
            <li><a href="<?php echo esc_url( home_url( '/' ) ); ?>" class="hover:text-[var(--wz-primary)] transition">Trang chủ</a></li>
            <li><span class="material-symbols-outlined text-[14px]" aria-hidden="true">chevron_right</span></li>
            <li><a href="<?php echo esc_url( home_url( '/my-account/' ) ); ?>" class="hover:text-[var(--wz-primary)] transition">Tài khoản</a></li>
            <li><span class="material-symbols-outlined text-[14px]" aria-hidden="true">chevron_right</span></li>
            <li class="font-bold text-[var(--wz-primary)]" aria-current="page">Thông báo</li>
        </ol>
    </nav>

    <div class="flex flex-col lg:flex-row lg:gap-6">
        <?php include get_stylesheet_directory() . '/template-parts/account/sidebar.php'; ?>

        <div class="flex-1 min-w-0 mt-4 lg:mt-0 space-y-4">
            <div class="flex items-center justify-between">
                <div class="flex items-center gap-2">
                    <h1 class="text-xl font-bold text-[var(--wz-primary)]">Thông báo</h1>
                    <?php if ( $unread_count > 0 ) : ?>
                        <span class="bg-[var(--wz-error)] text-white text-xs font-bold rounded-full px-2 py-0.5"><?php echo $unread_count; ?></span>
                    <?php endif; ?>
                </div>
                <?php if ( $unread_count > 0 ) : ?>
                    <button type="button" id="wz-mark-all-read" class="text-sm text-[var(--wz-primary)] font-semibold hover:underline">Đọc tất cả</button>
                <?php endif; ?>
            </div>

            <!-- Filter Tabs -->
            <div class="flex gap-2 overflow-x-auto pb-1">
                <button type="button" class="wz-notif-tab bg-[var(--wz-primary)] text-white px-3 py-1.5 rounded-full text-xs font-semibold" data-filter="all">Tất cả</button>
                <button type="button" class="wz-notif-tab bg-[var(--wz-surface-cool)] text-[var(--wz-primary)] px-3 py-1.5 rounded-full text-xs font-semibold hover:bg-[var(--wz-primary)]/10" data-filter="unread">Chưa đọc</button>
                <button type="button" class="wz-notif-tab bg-[var(--wz-surface-cool)] text-[var(--wz-primary)] px-3 py-1.5 rounded-full text-xs font-semibold hover:bg-[var(--wz-primary)]/10" data-filter="order">Đơn hàng</button>
                <button type="button" class="wz-notif-tab bg-[var(--wz-surface-cool)] text-[var(--wz-primary)] px-3 py-1.5 rounded-full text-xs font-semibold hover:bg-[var(--wz-primary)]/10" data-filter="promo">Khuyến mãi</button>
                <button type="button" class="wz-notif-tab bg-[var(--wz-surface-cool)] text-[var(--wz-primary)] px-3 py-1.5 rounded-full text-xs font-semibold hover:bg-[var(--wz-primary)]/10" data-filter="system">Hệ thống</button>
            </div>

            <!-- Notification List -->
            <?php if ( empty( $notifications ) ) : ?>
                <div class="bg-[var(--wz-surface-pure)] rounded-[var(--wz-radius-card)] p-8 shadow-[0_4px_24px_rgba(27,58,122,0.08)] text-center">
                    <span class="material-symbols-outlined text-[48px] text-[var(--wz-outline)]/30" aria-hidden="true">notifications_none</span>
                    <p class="text-base font-bold text-[var(--wz-primary)] mt-3">Không có thông báo nào</p>
                </div>
            <?php else : ?>
                <div class="space-y-2" id="wz-notif-list">
                    <?php foreach ( $notifications as $notif ) :
                        $type  = $notif['type'] ?? 'system';
                        $icons = $type_icons[ $type ] ?? $type_icons['system'];
                        $is_read = ! empty( $notif['is_read'] );
                    ?>
                        <div class="wz-notif-item flex gap-3 p-4 rounded-[var(--wz-radius-card)] border transition hover:bg-[var(--wz-surface-cool)]/50 <?php echo $is_read ? 'bg-[var(--wz-surface-pure)] border-[var(--wz-outline)]/10' : 'bg-[var(--wz-primary)]/5 border-[var(--wz-primary)]/20'; ?>"
                             data-type="<?php echo esc_attr( $type ); ?>" data-read="<?php echo $is_read ? '1' : '0'; ?>">
                            <div class="w-10 h-10 rounded-full flex items-center justify-center shrink-0 <?php echo esc_attr( $icons['bg'] ); ?>">
                                <span class="material-symbols-outlined text-[20px] <?php echo esc_attr( $icons['color'] ); ?>" aria-hidden="true"><?php echo esc_html( $icons['icon'] ); ?></span>
                            </div>
                            <div class="flex-1 min-w-0">
                                <p class="text-sm font-semibold text-[var(--wz-on-surface)]"><?php echo esc_html( $notif['title'] ?? '' ); ?></p>
                                <p class="text-xs text-[var(--wz-outline)] mt-0.5 line-clamp-2"><?php echo esc_html( $notif['message'] ?? '' ); ?></p>
                                <p class="text-[10px] text-[var(--wz-outline)] mt-1"><?php echo esc_html( $notif['time_ago'] ?? '' ); ?></p>
                            </div>
                            <?php if ( ! $is_read ) : ?>
                                <span class="w-2 h-2 rounded-full bg-[var(--wz-primary)] shrink-0 mt-2"></span>
                            <?php endif; ?>
                        </div>
                    <?php endforeach; ?>
                </div>
            <?php endif; ?>
        </div>
    </div>

</div>
</div>

<?php get_footer(); ?>
```
