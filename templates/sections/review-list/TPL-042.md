## Code — PHP

```php
<!-- reviews.php -->
<?php
/**
 * Account — Reviews page.
 */
declare(strict_types=1);

if ( ! defined( 'ABSPATH' ) ) exit;

get_header();

if ( ! is_user_logged_in() ) {
	wp_safe_redirect( wp_login_url( function_exists( 'wz_get_myaccount_url' ) ? wz_get_myaccount_url() . 'reviews/' : home_url( '/my-account/reviews/' ) ) );
	exit;
}

$account_url = function_exists( 'wz_get_myaccount_url' ) ? wz_get_myaccount_url() : home_url( '/my-account/' );
$user_id     = get_current_user_id();

$reviews = get_comments( [
	'user_id' => $user_id,
	'type'    => 'review',
	'status'  => 'approve',
	'number'  => 20,
	'orderby' => 'comment_date',
	'order'   => 'DESC',
] );
?>

<main class="py-6 px-4 max-w-3xl mx-auto">
	<nav class="text-sm text-[var(--wz-text-muted)] mb-4" aria-label="Breadcrumb">
		<a href="<?php echo esc_url( home_url( '/' ) ); ?>" class="hover:underline"><?php esc_html_e( 'Trang chủ', 'wezone-haven' ); ?></a>
		<span class="mx-1">/</span>
		<a href="<?php echo esc_url( $account_url ); ?>" class="hover:underline"><?php esc_html_e( 'Tài khoản', 'wezone-haven' ); ?></a>
		<span class="mx-1">/</span>
		<span aria-current="page"><?php esc_html_e( 'Đánh giá', 'wezone-haven' ); ?></span>
	</nav>

	<h1 class="text-xl font-semibold text-[var(--wz-text)] mb-6"><?php esc_html_e( 'Đánh giá của tôi', 'wezone-haven' ); ?></h1>

	<?php if ( ! empty( $reviews ) ) : ?>
		<div class="space-y-4">
			<?php foreach ( $reviews as $review ) :
				$product_id = (int) $review->comment_post_ID;
				$rating     = (int) get_comment_meta( $review->comment_ID, 'rating', true );
				$product_title = get_the_title( $product_id );
				$product_url   = get_permalink( $product_id );
				$product_image = get_the_post_thumbnail_url( $product_id, 'thumbnail' );
			?>
				<div class="border border-[var(--wz-border)] rounded-lg p-4">
					<div class="flex gap-3 items-start">
						<a href="<?php echo esc_url( $product_url ); ?>" class="shrink-0 w-14 h-14 rounded overflow-hidden bg-[var(--wz-surface)]">
							<?php if ( $product_image ) : ?>
								<img src="<?php echo esc_url( $product_image ); ?>" alt="<?php echo esc_attr( $product_title ?? '' ); ?>" class="w-full h-full object-cover" loading="lazy">
							<?php endif; ?>
						</a>
						<div class="flex-1 min-w-0">
							<a href="<?php echo esc_url( $product_url ); ?>" class="text-sm font-medium text-[var(--wz-text)] hover:underline">
								<?php echo esc_html( $product_title ); ?>
							</a>
							<?php if ( $rating > 0 ) : ?>
								<div class="flex items-center gap-0.5 mt-1">
									<?php for ( $i = 1; $i <= 5; $i++ ) : ?>
										<span aria-hidden="true" class="material-symbols-outlined text-sm <?php echo $i <= $rating ? 'text-yellow-500' : 'text-gray-300'; ?>" style="font-variation-settings: 'FILL' <?php echo $i <= $rating ? '1' : '0'; ?>">star</span>
									<?php endfor; ?>
								</div>
							<?php endif; ?>
							<p class="text-sm text-[var(--wz-text-muted)] mt-1"><?php echo esc_html( $review->comment_content ); ?></p>
							<span class="text-xs text-[var(--wz-text-muted)] mt-1 block"><?php echo esc_html( date_i18n( 'd/m/Y', strtotime( $review->comment_date ) ) ); ?></span>
						</div>
					</div>
				</div>
			<?php endforeach; ?>
		</div>
	<?php else : ?>
		<div class="flex flex-col items-center justify-center py-16 text-center">
			<span aria-hidden="true" class="material-symbols-outlined text-5xl text-[var(--wz-text-muted)] mb-4">rate_review</span>
			<h2 class="text-lg font-semibold text-[var(--wz-text)] mb-2">
				<?php esc_html_e( 'Chưa có đánh giá nào', 'wezone-haven' ); ?>
			</h2>
			<p class="text-sm text-[var(--wz-text-muted)] max-w-md">
				<?php esc_html_e( 'Sau khi mua hàng, bạn có thể đánh giá sản phẩm để giúp người mua khác tham khảo.', 'wezone-haven' ); ?>
			</p>
		</div>
	<?php endif; ?>
</main>

<?php get_footer(); ?>
```
