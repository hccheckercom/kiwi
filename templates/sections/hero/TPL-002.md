## Preview

Hero toàn màn hình với ảnh nền từ admin, gradient overlay tối dần từ trái. Content bên trái: eyebrow nhỏ uppercase → title serif italic lớn → mô tả → CTA button. Dưới cùng: 3 product cards "Shop the Look" glassmorphism (hiện trên mọi breakpoint). Admin quản lý toàn bộ nội dung banner từ dashboard.

## Architecture

```
Admin (wp-admin → Marketing → Banner)
  ↓ saves to wp_wz_banners table
  ↓
Theme calls wz_get_banners('home_hero')
  ↓
Render based on layout setting (wz_get_banner_layout())
  ├── classic: 1 hero + product cards
  ├── slider: multiple slides with dots
  └── mini-left: 2 mini banners + 1 main
```

## Code — PHP (inc/banner-helpers.php)

```php
<?php
declare(strict_types=1);

if ( ! defined( 'ABSPATH' ) ) { exit; }

if ( ! function_exists( 'wz_get_banners' ) ) :
function wz_get_banners( string $position ): array {
    global $wpdb;
    $table = $wpdb->prefix . 'wz_banners';
    $now   = current_time( 'mysql' );

    $results = $wpdb->get_results(
        $wpdb->prepare(
            "SELECT * FROM {$table}
            WHERE position = %s AND status = 'active'
            AND (start_date IS NULL OR start_date <= %s)
            AND (end_date IS NULL OR end_date >= %s)
            ORDER BY sort_order ASC",
            $position, $now, $now
        ), ARRAY_A
    );

    if ( ! $results ) { return []; }

    foreach ( $results as &$banner ) {
        if ( ! empty( $banner['image_id'] ) ) {
            $banner['image_url'] = wp_get_attachment_image_url( (int) $banner['image_id'], 'full' );
        }
        if ( ! empty( $banner['meta'] ) ) {
            $banner['meta'] = json_decode( $banner['meta'], true );
        } else {
            $banner['meta'] = null;
        }
        if ( ! empty( $banner['reference_id'] ) ) {
            $banner['reference_data'] = wz_get_banner_reference( $banner['type'], (int) $banner['reference_id'] );
        }
    }
    unset( $banner );
    return $results;
}
endif;

if ( ! function_exists( 'wz_get_banner_reference' ) ) :
function wz_get_banner_reference( string $type, int $id ): ?array {
    global $wpdb;
    if ( 'product' === $type ) {
        $product = $wpdb->get_row( $wpdb->prepare(
            "SELECT id, name, slug, price, sale_price, regular_price, image_id
            FROM {$wpdb->prefix}wz_products WHERE id = %d AND status = 'publish'", $id
        ), ARRAY_A );
        if ( $product ) {
            $product['image_url'] = ! empty( $product['image_id'] ) ? wp_get_attachment_image_url( (int) $product['image_id'], 'medium' ) : '';
            $product['permalink'] = home_url( '/product/' . $product['slug'] . '/' );
        }
        return $product;
    }
    if ( 'category' === $type ) {
        $category = $wpdb->get_row( $wpdb->prepare(
            "SELECT id, name, slug, image_id, count FROM {$wpdb->prefix}wz_categories WHERE id = %d", $id
        ), ARRAY_A );
        if ( $category ) {
            $category['image_url'] = ! empty( $category['image_id'] ) ? wp_get_attachment_image_url( (int) $category['image_id'], 'medium' ) : '';
            $category['url'] = home_url( '/?cat=' . $category['slug'] );
        }
        return $category;
    }
    return null;
}
endif;

if ( ! function_exists( 'wz_get_banner_layout' ) ) :
function wz_get_banner_layout(): string {
    return get_option( 'wz_banner_layout', 'classic' );
}
endif;
```

## Code — PHP (template-parts/hero/default.php)

```php
<?php
$shop_url     = function_exists( 'wz_get_shop_url' ) ? wz_get_shop_url() : home_url( '/shop/' );
$layout       = function_exists( 'wz_get_banner_layout' ) ? wz_get_banner_layout() : 'classic';
$hero_banners = function_exists( 'wz_get_banners' ) ? wz_get_banners( 'home_hero' ) : [];

$fallback = [
    'image_url'  => file_exists( WEZONE_HAVEN_DIR . '/assets/images/hero-banner.jpg' )
        ? WEZONE_HAVEN_URL . '/assets/images/hero-banner.jpg' : '',
    'title'      => __( 'Bộ sưu tập Tối Giản 2025', 'wezone-haven' ),
    'subtitle'   => __( 'Mô tả ngắn về bộ sưu tập.', 'wezone-haven' ),
    'badge_text' => __( 'Bộ sưu tập', 'wezone-haven' ),
    'cta_text'   => __( 'Xem bộ sưu tập', 'wezone-haven' ),
    'link_url'   => $shop_url,
];

if ( empty( $hero_banners ) ) {
    $hero_banners = [ $fallback ];
}

// Featured products: from banner meta OR fallback query
$hero_products = [];
$first_meta    = $hero_banners[0]['meta'] ?? null;
if ( ! empty( $first_meta['featured_products'] ) ) {
    $hero_products = $first_meta['featured_products'];
} elseif ( function_exists( 'wz_get_products' ) ) {
    $products = wz_get_products( [ 'limit' => 3, 'orderby' => 'created_at', 'order' => 'DESC' ] );
    foreach ( $products as $p ) {
        $hero_products[] = [
            'name'  => $p['name'],
            'price' => number_format( (float) $p['price'], 0, ',', '.' ) . '₫',
            'image' => ! empty( $p['image_id'] ) ? wp_get_attachment_image_url( (int) $p['image_id'], 'medium' ) : '',
            'link'  => home_url( '/product/' . $p['slug'] . '/' ),
        ];
    }
}

// Render function for reuse across layouts
function haven_render_hero_slide( array $banner, array $products, string $shop_url, bool $show_products = true ): void {
    $img   = $banner['image_url'] ?? '';
    $title = $banner['title'] ?? '';
    $sub   = $banner['subtitle'] ?? '';
    $badge = $banner['badge_text'] ?? '';
    $cta   = $banner['cta_text'] ?? __( 'Xem ngay', 'wezone-haven' );
    $link  = $banner['link_url'] ?? $shop_url;
    ?>
    <div class="haven-hero__visual">
        <?php if ( $img ) : ?>
            <img src="<?php echo esc_url( $img ); ?>" alt="" aria-hidden="true" class="haven-hero__img" />
        <?php endif; ?>
        <div class="haven-hero__overlay"></div>

        <div class="haven-hero__content">
            <?php if ( $badge ) : ?>
            <p class="haven-hero__eyebrow"><?php echo esc_html( $badge ); ?></p>
            <?php endif; ?>
            <h1 class="haven-hero__title"><?php echo esc_html( $title ); ?></h1>
            <?php if ( $sub ) : ?>
            <p class="haven-hero__desc"><?php echo esc_html( $sub ); ?></p>
            <?php endif; ?>
            <div>
                <a href="<?php echo esc_url( $link ); ?>" class="haven-hero__cta">
                    <?php echo esc_html( $cta ); ?>
                    <span class="material-symbols-outlined" aria-hidden="true" style="font-size:16px;">arrow_forward</span>
                </a>
            </div>
        </div>

        <?php if ( $show_products && ! empty( $products ) ) : ?>
        <div class="haven-hero__products">
            <?php foreach ( $products as $product ) : ?>
                <a href="<?php echo esc_url( $product['link'] ?? $shop_url ); ?>" class="haven-hero-card">
                    <div class="haven-hero-card__img-wrap">
                        <?php if ( ! empty( $product['image'] ) ) : ?>
                            <img src="<?php echo esc_url( $product['image'] ); ?>" alt="<?php echo esc_attr( $product['name'] ); ?>" loading="lazy" />
                        <?php else : ?>
                            <div class="haven-hero-card__placeholder">
                                <span class="material-symbols-outlined" aria-hidden="true">image</span>
                            </div>
                        <?php endif; ?>
                    </div>
                    <div class="haven-hero-card__info">
                        <span class="haven-hero-card__name"><?php echo esc_html( $product['name'] ); ?></span>
                        <span class="haven-hero-card__price"><?php echo esc_html( $product['price'] ); ?></span>
                    </div>
                </a>
            <?php endforeach; ?>
        </div>
        <?php endif; ?>
    </div>
    <?php
}

// Layout switch
if ( 'slider' === $layout ) : ?>
<section class="haven-hero haven-hero--slider" data-slider>
    <div class="haven-hero__track" data-slider-track>
        <?php foreach ( $hero_banners as $i => $banner ) : ?>
            <div class="haven-hero__slide<?php echo 0 === $i ? ' is-active' : ''; ?>" data-slide="<?php echo esc_attr( $i ); ?>">
                <?php haven_render_hero_slide( $banner, $hero_products, $shop_url, 0 === $i ); ?>
            </div>
        <?php endforeach; ?>
    </div>
    <?php if ( count( $hero_banners ) > 1 ) : ?>
    <div class="haven-hero__dots">
        <?php foreach ( $hero_banners as $i => $b ) : ?>
            <button class="haven-hero__dot<?php echo 0 === $i ? ' is-active' : ''; ?>" data-dot="<?php echo esc_attr( $i ); ?>"></button>
        <?php endforeach; ?>
    </div>
    <?php endif; ?>
</section>
<?php elseif ( 'mini-left' === $layout ) : ?>
<section class="haven-hero haven-hero--mini-left">
    <div class="haven-hero__mini-grid">
        <?php
        $mini_banners = array_merge( wz_get_banners( 'home_mini_1' ), wz_get_banners( 'home_mini_2' ) );
        $mini_left    = array_slice( $mini_banners, 0, 2 );
        $main_banner  = $hero_banners[0] ?? $fallback;
        ?>
        <div class="haven-hero__mini-col">
            <?php foreach ( $mini_left as $mini ) : ?>
            <a href="<?php echo esc_url( $mini['link_url'] ?? $shop_url ); ?>" class="haven-hero__mini-item">
                <?php if ( ! empty( $mini['image_url'] ) ) : ?>
                    <img src="<?php echo esc_url( $mini['image_url'] ); ?>" alt="<?php echo esc_attr( $mini['title'] ?? '' ); ?>" loading="lazy" />
                <?php endif; ?>
            </a>
            <?php endforeach; ?>
        </div>
        <div class="haven-hero__main-col">
            <?php haven_render_hero_slide( $main_banner, $hero_products, $shop_url ); ?>
        </div>
    </div>
</section>
<?php else : // classic ?>
<section class="haven-hero">
    <?php haven_render_hero_slide( $hero_banners[0], $hero_products, $shop_url ); ?>
</section>
<?php endif; ?>
```

## Featured Products — Data Flow

```
1. Admin tạo banner → meta JSON chứa featured_products:
   {
     "featured_products": [
       { "name": "Sofa Nordic", "price": "12.900.000₫", "image": "url", "link": "/product/sofa-nordic/" },
       ...
     ]
   }

2. Nếu admin không set featured_products → fallback:
   wz_get_products(['limit' => 3, 'orderby' => 'created_at', 'order' => 'DESC'])

3. Cards render dạng glassmorphism overlay trên ảnh hero
```

## Usage Notes

- Banner content 100% dynamic từ admin — KHÔNG hardcode
- `inc/banner-helpers.php` là shim — khi wezone-core active, plugin cung cấp function gốc
- Product cards hiện trên MỌI breakpoint (mobile nhỏ hơn, tablet/desktop lớn hơn)
- Mobile: banner cao hơn (620px) + padding-bottom 180px để cards không che CTA
- 3 layouts admin chọn: classic (mặc định), slider, mini-left
- Cần `require_once` banner-helpers.php trong functions.php
