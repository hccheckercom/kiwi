<!-- compare.php -->
<?php
/**
 * WeZone template: So sánh sản phẩm (Compare).
 *
 * @package WeZone\TrungAnhV2
 */

declare(strict_types=1);

if ( ! defined( 'ABSPATH' ) ) {
	exit;
}

get_header();

$compare_ids = [];
if ( ! empty( $_COOKIE['wz_compare_ids'] ) ) {
	$raw = json_decode( wp_unslash( $_COOKIE['wz_compare_ids'] ), true );
	if ( is_array( $raw ) ) {
		$compare_ids = array_map( 'intval', array_slice( $raw, 0, 4 ) );
	}
}

$products = [];
foreach ( $compare_ids as $pid ) {
	if ( function_exists( 'wz_get_product' ) ) {
		$p = wz_get_product( $pid );
		if ( is_array( $p ) && ! empty( $p['id'] ) ) {
			$attrs = [];
			if ( function_exists( 'wz_get_product_attributes' ) ) {
				$raw_attrs = wz_get_product_attributes( $pid );
				if ( is_array( $raw_attrs ) ) {
					foreach ( $raw_attrs as $attr ) {
						if ( ! empty( $attr['label'] ) && isset( $attr['value'] ) ) {
							$attrs[ $attr['label'] ] = $attr['value'];
						}
					}
				}
			}
			$p['_attrs']  = $attrs;
			$p['_thumb']  = wp_get_attachment_image_url( (int) ( $p['image_id'] ?? get_post_thumbnail_id( $pid ) ), 'medium' );
			$p['_rating'] = (float) get_post_meta( $pid, '_wz_average_rating', true );
			$p['_count']  = (int) get_post_meta( $pid, '_wz_review_count', true );
			$products[]   = $p;
		}
	}
}

$all_labels = [];
foreach ( $products as $p ) {
	foreach ( array_keys( $p['_attrs'] ) as $label ) {
		if ( ! in_array( $label, $all_labels, true ) ) {
			$all_labels[] = $label;
		}
	}
}

$fmt = function_exists( 'wz_format_price' ) ? 'wz_format_price' : fn( $v ) => number_format( (float) $v, 0, ',', '.' ) . 'đ';
$shop_url = function_exists( 'wz_get_shop_url' ) ? wz_get_shop_url() : home_url( '/' );
?>

<main id="main-content" class="max-w-container-max mx-auto px-lg py-md pb-28">

	<nav class="flex items-center gap-sm text-on-surface-variant text-label-caps mb-gutter flex-wrap">
		<a href="<?php echo esc_url( home_url( '/' ) ); ?>" class="hover:text-primary transition-colors">Trang chủ</a>
		<?php wz_icon( 'chevron_right', [ 'size' => 'xs', 'color' => 'muted' ] ); ?>
		<span class="font-bold text-on-surface">So sánh sản phẩm</span>
	</nav>

	<h1 class="text-2xl font-bold mb-6" style="color:var(--color-text-primary)">So sánh sản phẩm</h1>

	<?php if ( empty( $products ) ) : ?>
		<?php get_template_part( 'template-parts/components/empty-state', null, [ 'context' => 'compare' ] ); ?>
		<div class="text-center mt-6">
			<a href="<?php echo esc_url( $shop_url ); ?>" class="inline-flex items-center gap-2 px-6 py-3 rounded-xl font-bold text-white transition-all" style="background:var(--color-primary)">
				Khám phá sản phẩm
			</a>
		</div>
	<?php else : ?>
		<div class="wz-spec-table-wrapper" style="margin-bottom:var(--space-8)">
			<table class="wz-spec-table" style="min-width:<?php echo esc_attr( (string) ( 120 + count( $products ) * 180 ) ); ?>px">
				<thead>
					<tr>
						<th style="position:sticky;left:0;z-index:2;background:var(--color-bg-section);min-width:120px"></th>
						<?php foreach ( $products as $p ) : ?>
							<th style="text-align:center;min-width:180px;vertical-align:top;padding:var(--space-4)">
								<?php if ( ! empty( $p['_thumb'] ) ) : ?>
									<img src="<?php echo esc_url( $p['_thumb'] ); ?>" alt="<?php echo esc_attr( $p['name'] ?? '' ); ?>" style="width:80px;height:80px;object-fit:contain;margin:0 auto var(--space-2);display:block;border-radius:var(--radius-md);background:var(--color-surface-alt)">
								<?php endif; ?>
								<a href="<?php echo esc_url( home_url( '/product/' . ( $p['slug'] ?? '' ) . '/' ) ); ?>" class="block font-bold text-sm hover:text-primary transition-colors" style="color:var(--color-text-primary);line-height:1.3">
									<?php echo esc_html( $p['name'] ?? '' ); ?>
								</a>
							</th>
						<?php endforeach; ?>
					</tr>
				</thead>
				<tbody>
					<!-- Giá -->
					<tr>
						<td style="position:sticky;left:0;z-index:1;background:var(--color-surface);font-weight:var(--font-semibold);font-size:var(--text-sm);color:var(--color-text-secondary)">Giá</td>
						<?php foreach ( $products as $p ) : ?>
							<td style="text-align:center;font-weight:var(--font-bold);color:var(--color-primary)">
								<?php echo esc_html( $fmt( $p['sale_price'] ?? $p['price'] ?? 0 ) ); ?>
							</td>
						<?php endforeach; ?>
					</tr>
					<!-- Đánh giá -->
					<tr>
						<td style="position:sticky;left:0;z-index:1;background:var(--color-surface);font-weight:var(--font-semibold);font-size:var(--text-sm);color:var(--color-text-secondary)">Đánh giá</td>
						<?php foreach ( $products as $p ) : ?>
							<td style="text-align:center;font-size:var(--text-sm)">
								<?php if ( $p['_rating'] > 0 ) : ?>
									<span style="color:var(--color-accent)">★</span> <?php echo esc_html( number_format( $p['_rating'], 1 ) ); ?> (<?php echo esc_html( (string) $p['_count'] ); ?>)
								<?php else : ?>
									—
								<?php endif; ?>
							</td>
						<?php endforeach; ?>
					</tr>
					<!-- Tình trạng -->
					<tr>
						<td style="position:sticky;left:0;z-index:1;background:var(--color-surface);font-weight:var(--font-semibold);font-size:var(--text-sm);color:var(--color-text-secondary)">Tình trạng</td>
						<?php foreach ( $products as $p ) :
							$in_stock = ! empty( $p['in_stock'] ) || ( isset( $p['stock_status'] ) && 'outofstock' !== $p['stock_status'] );
							?>
							<td style="text-align:center;font-size:var(--text-sm);color:<?php echo $in_stock ? 'var(--color-success, #059669)' : 'var(--color-error, #dc2626)'; ?>">
								<?php echo $in_stock ? 'Còn hàng' : 'Hết hàng'; ?>
							</td>
						<?php endforeach; ?>
					</tr>
					<!-- Dynamic attributes -->
					<?php foreach ( $all_labels as $label ) : ?>
						<tr>
							<td style="position:sticky;left:0;z-index:1;background:var(--color-surface);font-weight:var(--font-semibold);font-size:var(--text-sm);color:var(--color-text-secondary)">
								<?php echo esc_html( $label ); ?>
							</td>
							<?php foreach ( $products as $p ) : ?>
								<td style="text-align:center;font-size:var(--text-sm)">
									<?php echo esc_html( $p['_attrs'][ $label ] ?? '—' ); ?>
								</td>
							<?php endforeach; ?>
						</tr>
					<?php endforeach; ?>
					<!-- Actions -->
					<tr>
						<td style="position:sticky;left:0;z-index:1;background:var(--color-surface)"></td>
						<?php foreach ( $products as $p ) : ?>
							<td style="text-align:center;padding:var(--space-4)">
								<a href="<?php echo esc_url( home_url( '/product/' . ( $p['slug'] ?? '' ) . '/' ) ); ?>" class="inline-flex items-center justify-center gap-1 px-4 py-2 rounded-lg font-bold text-sm text-white transition-all" style="background:var(--color-primary)">
									Xem sản phẩm
								</a>
								<button type="button" class="wz-compare-remove block mx-auto mt-2 text-xs transition-colors hover:text-primary" style="color:var(--color-text-muted)" data-id="<?php echo esc_attr( (string) $p['id'] ); ?>">
									Xóa khỏi so sánh
								</button>
							</td>
						<?php endforeach; ?>
					</tr>
				</tbody>
			</table>
		</div>
	<?php endif; ?>

</main>

<script>
document.querySelectorAll('.wz-compare-remove').forEach(function(btn) {
	btn.addEventListener('click', function() {
		var id = parseInt(this.dataset.id, 10);
		var ids = [];
		try { ids = JSON.parse(document.cookie.replace(/(?:(?:^|.*;\s*)wz_compare_ids\s*=\s*([^;]*).*$)|^.*$/, '$1') || '[]'); } catch(e) {}
		ids = ids.filter(function(v) { return v !== id; });
		document.cookie = 'wz_compare_ids=' + JSON.stringify(ids) + ';path=/;max-age=2592000';
		location.reload();
	});
});
</script>

<?php
get_footer();
