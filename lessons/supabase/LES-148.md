## Bad
```tsx
useEffect(() => {
  const channel = supabase.channel('orders')
    .on('postgres_changes', { event: '*', schema: 'public', table: 'orders' }, (payload) => {
      setOrders(prev => [...prev, payload.new])
    })
    .subscribe()
  // No cleanup — channel stays open after unmount
}, [])
```

## Good
```tsx
useEffect(() => {
  const channel = supabase.channel('orders')
    .on('postgres_changes', { event: '*', schema: 'public', table: 'orders' }, (payload) => {
      setOrders(prev => [...prev, payload.new])
    })
    .subscribe()

  return () => {
    supabase.removeChannel(channel)
  }
}, [])
```

## Also applies to: feedback-draw realtime (lib/feedback-draw/realtime.ts)
```ts
// Pattern: create channel → subscribe → return unsubscribe function
export function subscribeToAnnotations(projectId: string, onUpdate: Callback) {
  const channel = supabase.channel(`annotations:${projectId}`)
    .on('postgres_changes', { ... }, onUpdate)
    .subscribe()
  return () => supabase.removeChannel(channel) // Caller must invoke on cleanup
}
```

## Why
Supabase Realtime channels maintain WebSocket connections. Without cleanup, each navigation creates a new subscription while old ones stay alive. This causes: memory leaks, duplicate event handlers (data appears twice), and eventually hits Supabase's connection limit (100 per client).
