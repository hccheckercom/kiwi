## Bad
```ts
export async function POST(req: Request) {
  const { prompt } = await req.json()
  // User closes tab → this keeps running → wastes API credits
  const response = await anthropic.messages.create({
    model: 'claude-sonnet-4-6',
    max_tokens: 4096,
    messages: [{ role: 'user', content: prompt }],
  })
  return Response.json({ result: response.content[0].text })
}
```

## Good
```ts
export async function POST(req: Request) {
  const { prompt } = await req.json()
  const controller = new AbortController()

  // Cancel AI call if client disconnects
  req.signal.addEventListener('abort', () => controller.abort())

  try {
    const response = await anthropic.messages.create({
      model: 'claude-sonnet-4-6',
      max_tokens: 4096,
      messages: [{ role: 'user', content: prompt }],
    }, { signal: controller.signal })
    return Response.json({ result: response.content[0].text })
  } catch (err) {
    if (controller.signal.aborted) return new Response(null, { status: 499 })
    throw err
  }
}
```

## Why
Real bug: commit 61393fe. GMC AI scan runs multiple Claude calls (5-10 per scan). Without AbortController, navigating away or starting a new scan leaves old calls running — burning API credits. Always wire req.signal to AI client abort.
