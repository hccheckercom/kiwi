# FAQ Accordion — Collapsible Q&A Widget

Accordion-style FAQ widget với smooth expand/collapse animation.

## Use Cases
- FAQ page (trang 36)
- Help center
- Product support section
- Policy clarifications

## Features
- 🎯 Accordion expand/collapse (one at a time)
- ➕ Plus/minus icon toggle
- 🎨 Smooth height animation
- 📱 Mobile-friendly tap targets
- ♿ Keyboard accessible (Enter/Space)
- 🔍 SEO-friendly (all content in HTML)

## Code

### PHP Template (template-parts/faq/accordion.php)

```php
<?php
/**
 * FAQ Accordion Component
 * 
 * @param array $faqs Array of FAQ items [question, answer]
 */

$faqs = $args['faqs'] ?? [];
if ( empty( $faqs ) ) {
	return;
}
?>

<div class="faq-accordion space-y-4">
	<?php foreach ( $faqs as $index => $faq ): ?>
		<div class="faq-item bg-white rounded-lg border border-gray-200 overflow-hidden">
			<button 
				type="button"
				class="faq-question w-full px-6 py-4 flex items-center justify-between text-left hover:bg-gray-50 transition-colors"
				aria-expanded="false"
				aria-controls="faq-answer-<?php echo esc_attr( $index ); ?>"
			>
				<span class="font-semibold text-gray-900 pr-4"><?php echo esc_html( $faq['question'] ); ?></span>
				<svg class="faq-icon w-5 h-5 text-gray-500 shrink-0 transition-transform" fill="none" stroke="currentColor" viewBox="0 0 24 24">
					<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M12 4v16m8-8H4" />
				</svg>
			</button>
			<div 
				id="faq-answer-<?php echo esc_attr( $index ); ?>"
				class="faq-answer overflow-hidden transition-all duration-300"
				style="max-height: 0;"
				aria-hidden="true"
			>
				<div class="px-6 py-4 text-gray-600 border-t border-gray-100">
					<?php echo wp_kses_post( wpautop( $faq['answer'] ) ); ?>
				</div>
			</div>
		</div>
	<?php endforeach; ?>
</div>
```

### JavaScript (assets/js/faq-accordion.js)

```javascript
/**
 * FAQ Accordion Handler
 */
(function() {
	'use strict';

	function initFaqAccordion() {
		const accordion = document.querySelector('.faq-accordion');
		if (!accordion) return;

		const items = accordion.querySelectorAll('.faq-item');

		items.forEach(item => {
			const button = item.querySelector('.faq-question');
			const answer = item.querySelector('.faq-answer');
			const icon = item.querySelector('.faq-icon');

			if (!button || !answer) return;

			button.addEventListener('click', () => {
				const isExpanded = button.getAttribute('aria-expanded') === 'true';

				// Close all other items
				items.forEach(otherItem => {
					if (otherItem !== item) {
						const otherButton = otherItem.querySelector('.faq-question');
						const otherAnswer = otherItem.querySelector('.faq-answer');
						const otherIcon = otherItem.querySelector('.faq-icon');

						otherButton.setAttribute('aria-expanded', 'false');
						otherAnswer.style.maxHeight = '0';
						otherAnswer.setAttribute('aria-hidden', 'true');
						otherIcon.style.transform = 'rotate(0deg)';
					}
				});

				// Toggle current item
				if (isExpanded) {
					button.setAttribute('aria-expanded', 'false');
					answer.style.maxHeight = '0';
					answer.setAttribute('aria-hidden', 'true');
					icon.style.transform = 'rotate(0deg)';
				} else {
					button.setAttribute('aria-expanded', 'true');
					answer.style.maxHeight = answer.scrollHeight + 'px';
					answer.setAttribute('aria-hidden', 'false');
					icon.style.transform = 'rotate(45deg)'; // Plus → X
				}
			});

			// Keyboard support
			button.addEventListener('keydown', (e) => {
				if (e.key === 'Enter' || e.key === ' ') {
					e.preventDefault();
					button.click();
				}
			});
		});
	}

	// Init on DOM ready
	if (document.readyState === 'loading') {
		document.addEventListener('DOMContentLoaded', initFaqAccordion);
	} else {
		initFaqAccordion();
	}
})();
```

## Data Sources

**Static data** (hardcoded in template):
```php
$faqs = wz_config('faq_items', [
	['question' => '...', 'answer' => '...'],
	// ...
]);
```

**Dynamic from CPT** (if using FAQ post type):
```php
$faq_query = new WP_Query([
	'post_type' => 'wz_faq',
	'posts_per_page' => -1,
	'orderby' => 'menu_order',
	'order' => 'ASC',
]);

$faqs = [];
if ($faq_query->have_posts()) {
	while ($faq_query->have_posts()) {
		$faq_query->the_post();
		$faqs[] = [
			'question' => get_the_title(),
			'answer' => get_the_content(),
		];
	}
	wp_reset_postdata();
}
```

## Responsive Behavior
- Mobile: Full width, comfortable tap targets (py-4)
- Desktop: Same layout, hover states

## Accessibility
- ✅ `aria-expanded` state
- ✅ `aria-controls` linking
- ✅ `aria-hidden` on collapsed content
- ✅ Keyboard navigation (Enter/Space)
- ✅ Focus visible states
- ✅ Semantic button elements

## Animation Details
- Expand: `max-height: 0 → scrollHeight` (300ms ease)
- Icon rotate: `0deg → 45deg` (plus becomes X)
- Smooth height transition via inline style

```javascript
// Change rotation
icon.style.transform = isExpanded ? 'rotate(0deg)' : 'rotate(180deg)';
```

### Add search filter
```javascript
const searchInput = document.querySelector('#faq-search');
searchInput.addEventListener('input', (e) => {
	const query = e.target.value.toLowerCase();
	items.forEach(item => {
		const question = item.querySelector('.faq-question').textContent.toLowerCase();
		const answer = item.querySelector('.faq-answer').textContent.toLowerCase();
		item.style.display = (question.includes(query) || answer.includes(query)) ? 'block' : 'none';
	});
});
```
