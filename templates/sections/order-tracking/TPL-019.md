<?php
/**
 * Tracking page inline JavaScript — [28].

declare(strict_types=1);

<script>
(function() {
	var form      = document.getElementById('wz-tracking-form');
	var submitBtn = document.getElementById('wz-tracking-submit');
	var errorEl   = document.getElementById('wz-tracking-error');
	var errorMsg  = document.getElementById('wz-tracking-error-msg');
	var resultEl  = document.getElementById('wz-tracking-result');
	var statusMap = JSON.parse(document.getElementById('wz-status-map').textContent);

	if (!form) return;

	var ajaxUrl = (typeof wzTheme !== 'undefined') ? wzTheme.ajaxurl : '<?php echo esc_url( admin_url( 'admin-ajax.php' ) ); ?>';
	var nonce   = form.querySelector('[name="wz_tracking_nonce"]');

	form.addEventListener('submit', function(e) {
		e.preventDefault();

		var orderId = form.querySelector('[name="order_id"]').value.replace(/\D/g, '');
		var email   = form.querySelector('[name="order_email"]').value.trim();

		if (!orderId || !email) return;

		errorEl.classList.add('hidden');
		resultEl.classList.add('hidden');
		submitBtn.disabled = true;
		submitBtn.querySelector('span:first-child').textContent = 'hourglass_empty';

		var fd = new FormData();
		fd.append('action', 'wz_track_order');
		fd.append('nonce', nonce ? nonce.value : '');
		fd.append('order_id', orderId);
		fd.append('order_email', email);

		fetch(ajaxUrl, { method: 'POST', body: fd })
			.then(function(r) { return r.json(); })
			.then(function(res) {
				submitBtn.disabled = false;
				submitBtn.querySelector('span:first-child').textContent = 'search';

				if (!res.success) {
					errorMsg.textContent = res.data && res.data.message ? res.data.message : 'Không tìm thấy đơn hàng.';
					errorEl.classList.remove('hidden');
					return;
				}

				renderResult(res.data);
				resultEl.classList.remove('hidden');
			})
			.catch(function() {
				submitBtn.disabled = false;
				submitBtn.querySelector('span:first-child').textContent = 'search';
				errorMsg.textContent = 'Đã có lỗi xảy ra. Vui lòng thử lại.';
				errorEl.classList.remove('hidden');
			});
	});

	function renderResult(data) {
		var st = statusMap[data.status] || { label: data.status, class: 'bg-gray-100 text-gray-800' };
		document.getElementById('wz-track-order-code').textContent = '#WZ-' + data.id;
		document.getElementById('wz-track-date').textContent = data.date_created || '';

		var badge = document.getElementById('wz-track-badge');
		badge.textContent = st.label;
		badge.className = 'px-3 py-1 rounded-full text-xs font-semibold ' + st.class;

		renderTimeline(data.timeline || []);
		renderShipping(data);
		renderProducts(data.items || [], data.total_items || 0);
		renderAddress(data.shipping_address || {});
	}

	function renderTimeline(steps) {
		var desktop = document.getElementById('wz-track-timeline');
		var mobile  = document.getElementById('wz-track-timeline-mobile');
		desktop.innerHTML = '';
		mobile.innerHTML  = '';

		var icons = ['shopping_cart', 'verified', 'local_shipping', 'check_circle'];

		steps.forEach(function(step, i) {
			var iconName = icons[i] || 'circle';
			var isLast   = i === steps.length - 1;

			// Desktop step
			var dStep = document.createElement('div');
			dStep.className = 'flex-1 flex flex-col items-center text-center relative';
			var iconClass = step.done
				? 'text-success'
				: step.current
					? 'text-primary'
					: 'text-muted';
			var iconText = step.done ? 'check_circle' : iconName;
			var labelClass = step.current ? 'font-semibold text-primary' : step.done ? 'text-success' : 'text-muted';

			dStep.innerHTML =
				'<span class="material-symbols-outlined icon--lg ' + iconClass + (step.current ? ' animate-pulse' : '') + '" data-wz-icon="' + iconText + '" aria-hidden="true">' + iconText + '</span>' +
				'<span class="text-xs mt-1 ' + labelClass + '">' + escHtml(step.label) + '</span>' +
				(step.timestamp ? '<span class="text-[10px] text-muted mt-0.5">' + escHtml(step.timestamp) + '</span>' : '');

			desktop.appendChild(dStep);

			if (!isLast) {
				var connector = document.createElement('div');
				connector.className = 'flex-1 flex items-center pt-3';
				connector.innerHTML = '<div class="w-full h-0.5 ' + (step.done ? 'bg-success' : 'bg-gray-200') + '"></div>';
				desktop.appendChild(connector);
			}

			// Mobile step
			var mStep = document.createElement('div');
			mStep.className = 'flex gap-3' + (isLast ? '' : ' pb-4');
			var lineHtml = isLast ? '' : '<div class="w-0.5 flex-1 mx-auto mt-1 ' + (step.done ? 'bg-success' : 'bg-gray-200') + '"></div>';
			mStep.innerHTML =
				'<div class="flex flex-col items-center">' +
					'<span class="material-symbols-outlined icon--md ' + iconClass + (step.current ? ' animate-pulse' : '') + '" data-wz-icon="' + iconText + '" aria-hidden="true">' + iconText + '</span>' +
					lineHtml +
				'</div>' +
				'<div class="pb-2">' +
					'<p class="text-sm ' + labelClass + '">' + escHtml(step.label) + '</p>' +
					(step.timestamp ? '<p class="text-[10px] text-muted">' + escHtml(step.timestamp) + '</p>' : '') +
				'</div>';

			mobile.appendChild(mStep);
		});
	}

	function renderShipping(data) {
		var el = document.getElementById('wz-track-shipping');
		if (data.tracking_code) {
			document.getElementById('wz-track-carrier').textContent = data.carrier || 'Đơn vị vận chuyển';
			document.getElementById('wz-track-code').textContent = data.tracking_code;
			el.classList.remove('hidden');
		} else {
			el.classList.add('hidden');
		}
	}

	function renderProducts(items, total) {
		var container = document.getElementById('wz-track-items');
		var moreEl    = document.getElementById('wz-track-more-items');
		container.innerHTML = '';

		var show = items.slice(0, 3);
		show.forEach(function(item) {
			var row = document.createElement('div');
			row.className = 'flex items-center gap-3 py-2';
			var imgSrc = item.image || '';
			row.innerHTML =
				(imgSrc ? '<img src="' + escAttr(imgSrc) + '" class="rounded-lg object-cover w-12 h-12 bg-gray-100" alt="" width="48" height="48" onerror="this.onerror=null;this.src=\'<?php echo esc_url( get_template_directory_uri() . "/assets/images/placeholder-product.svg" ); ?>\'" />' : '<div class="w-12 h-12 rounded-lg bg-gray-100 flex items-center justify-center">' + <?php echo wp_json_encode( wz_icon( 'image', [ 'color' => 'muted', 'echo' => false ] ) ); ?> + '</div>') +
				'<div class="flex-1 min-w-0">' +
					'<p class="text-sm text-on-surface truncate">' + escHtml(item.name || '') + '</p>' +
					(item.variant ? '<p class="text-xs text-muted">' + escHtml(item.variant) + '</p>' : '') +
				'</div>' +
				'<div class="text-right shrink-0">' +
					'<p class="text-sm text-on-surface">x' + (item.qty || 1) + '</p>' +
					'<p class="text-sm font-semibold text-on-surface">' + escHtml(item.subtotal || '') + '</p>' +
				'</div>';
			container.appendChild(row);
		});

		if (total > 3) {
			moreEl.classList.remove('hidden');
			var moreBtn = document.getElementById('wz-track-show-all');
			moreBtn.textContent = 'Xem thêm ' + (total - 3) + ' sản phẩm';
		} else {
			moreEl.classList.add('hidden');
		}
	}

	function renderAddress(addr) {
		var el = document.getElementById('wz-track-addr-content');
		if (!addr.name && !addr.address) {
			document.getElementById('wz-track-address').classList.add('hidden');
			return;
		}
		document.getElementById('wz-track-address').classList.remove('hidden');
		var parts = [addr.name, addr.phone, addr.address].filter(Boolean);
		el.innerHTML = '<p class="font-semibold">' + escHtml(parts[0] || '') + '</p>' +
			(parts[1] ? '<p class="text-muted">' + escHtml(parts[1]) + '</p>' : '') +
			(parts[2] ? '<p class="text-muted">' + escHtml(parts[2]) + '</p>' : '');
	}

	// Copy tracking code
	var copyBtn = document.getElementById('wz-copy-tracking');
	if (copyBtn) {
		copyBtn.addEventListener('click', function() {
			var code = document.getElementById('wz-track-code').textContent;
			if (navigator.clipboard) {
				navigator.clipboard.writeText(code).then(function() {
					copyBtn.querySelector('span').textContent = 'check';
					setTimeout(function() {
						copyBtn.querySelector('span').textContent = 'content_copy';
					}, 1500);
				});
			}
		});
	}

	function escHtml(s) {
		var d = document.createElement('div');
		d.appendChild(document.createTextNode(s));
		return d.innerHTML;
	}

	function escAttr(s) {
		return s.replace(/&/g, '&amp;').replace(/"/g, '&quot;').replace(/'/g, '&#39;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
	}
})();
</script>

	function renderAddress(addr) {
		var el = document.getElementById('wz-track-addr-content');
		if (!addr.name && !addr.address) {
			document.getElementById('wz-track-address').classList.add('hidden');

		document.getElementById('wz-track-address').classList.remove('hidden');
		var parts = [addr.name, addr.phone, addr.address].filter(Boolean);
		el.innerHTML = '<p class="font-semibold">' + escHtml(parts[0] || '') + '</p>' +
			(parts[1] ? '<p class="text-muted">' + escHtml(parts[1]) + '</p>' : '') +

	function escHtml(s) {
		var d = document.createElement('div');
		d.appendChild(document.createTextNode(s));

	}

	function escAttr(s) {
		return s.replace(/&/g, '&amp;').replace(/"/g, '&quot;').replace(/'/g, '&#39;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
