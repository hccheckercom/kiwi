## Bad
```tsx
function PromptManager({ mode }: Props) {
  return (
    <div>
      {mode === 'builder' && (
        <BuilderPanel
          onChange={useCallback((v) => save(v), [])} // Hook inside conditional!
        />
      )}
    </div>
  )
}
// React Error #310: Rendered fewer hooks than expected
```

## Good
```tsx
function PromptManager({ mode }: Props) {
  const handleChange = useCallback((v) => save(v), []) // Top-level — always called

  return (
    <div>
      {mode === 'builder' && (
        <BuilderPanel onChange={handleChange} />
      )}
    </div>
  )
}
```

## Why
Real bug: commit 909c35f. Hooks inside `{condition && <JSX>}` blocks violate Rules of Hooks — React tracks hooks by call order. When condition changes, hook count changes → crash. Always declare hooks at component top level.
