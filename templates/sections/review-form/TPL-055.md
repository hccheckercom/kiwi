# Review Form — Product Review Submission

Review submission form với star rating, validation, và AJAX submit.

## Use Cases
- Product review form (trang 50)
- Service review
- Testimonial submission

## Features
- ⭐ Interactive star rating (1-5 stars)
- 📝 Review text with character count
- 📸 Optional image upload
- ✅ Client-side validation
- 🔄 AJAX submission (no page reload)
- 🔐 Nonce protection
- 📧 Email notification to admin

### JavaScript (assets/js/review-form.js)

```javascript
/**
 * Review Form Handler
 */
(function() {
	'use strict';

	const form = document.getElementById('wz-review-form');
	if (!form) return;

	const stars = form.querySelectorAll('.star');
	const ratingInput = document.getElementById('rating-input');
	const contentTextarea = document.getElementById('review-content');
	const charCount = document.getElementById('char-count');
	const submitButton = document.getElementById('submit-review');
	const responseDiv = document.getElementById('review-response');

	// Star rating handler
	stars.forEach(star => {
		star.addEventListener('click', () => {
			const rating = parseInt(star.dataset.rating);
			ratingInput.value = rating;

			// Update star colors
			stars.forEach((s, index) => {
				if (index < rating) {
					s.classList.remove('text-gray-300');
					s.classList.add('text-yellow-400');
				} else {
					s.classList.remove('text-yellow-400');
					s.classList.add('text-gray-300');
				}
			});

			// Hide error
			document.getElementById('rating-error').classList.add('hidden');
		});

		// Hover effect
		star.addEventListener('mouseenter', () => {
			const rating = parseInt(star.dataset.rating);
			stars.forEach((s, index) => {
				if (index < rating) {
					s.classList.add('text-yellow-400');
				}
			});
		});

		star.addEventListener('mouseleave', () => {
			const currentRating = parseInt(ratingInput.value);
			stars.forEach((s, index) => {
				if (index >= currentRating) {
					s.classList.remove('text-yellow-400');
					s.classList.add('text-gray-300');
				}
			});
		});
	});

	// Character count
	contentTextarea.addEventListener('input', () => {
		const length = contentTextarea.value.length;
		charCount.textContent = `${length}/1000 ký tự`;

		if (length < 50) {
			charCount.classList.add('text-red-500');
		} else {
			charCount.classList.remove('text-red-500');
		}
	});

	// Form submission
	form.addEventListener('submit', async (e) => {
		e.preventDefault();

		// Validation
		const rating = parseInt(ratingInput.value);
		if (rating === 0) {
			document.getElementById('rating-error').classList.remove('hidden');
			return;
		}

		const content = contentTextarea.value.trim();
		if (content.length < 50) {
			showResponse('error', 'Nội dung đánh giá phải có ít nhất 50 ký tự');
			return;
		}

		// Disable submit button
		submitButton.disabled = true;
		submitButton.textContent = 'Đang gửi...';

		// Prepare form data
		const formData = new FormData(form);

		try {
			const response = await fetch(wzTheme.ajaxUrl, {
				method: 'POST',
				body: formData,
			});

			const data = await response.json();

			if (data.success) {
				showResponse('success', data.data.message || 'Đánh giá của bạn đã được gửi thành công!');
				form.reset();
				ratingInput.value = 0;
				stars.forEach(s => {
					s.classList.remove('text-yellow-400');
					s.classList.add('text-gray-300');
				});
			} else {
				showResponse('error', data.data.message || 'Có lỗi xảy ra, vui lòng thử lại');
			}
		} catch (error) {
			showResponse('error', 'Không thể kết nối đến server');
		} finally {
			submitButton.disabled = false;
			submitButton.textContent = 'Gửi đánh giá';
		}
	});

	function showResponse(type, message) {
		responseDiv.classList.remove('hidden', 'bg-green-50', 'text-green-800', 'bg-red-50', 'text-red-800');
		
		if (type === 'success') {
			responseDiv.classList.add('bg-green-50', 'text-green-800');
		} else {
			responseDiv.classList.add('bg-red-50', 'text-red-800');
		}

		responseDiv.textContent = message;

		// Auto-hide after 5s
		setTimeout(() => {
			responseDiv.classList.add('hidden');
		}, 5000);
	}
})();
```

### AJAX Handler (functions.php)

```php
/**
 * Handle review submission
 */
add_action( 'wp_ajax_wz_submit_review', 'wz_handle_review_submission' );
add_action( 'wp_ajax_nopriv_wz_submit_review', 'wz_handle_review_submission' );

function wz_handle_review_submission() {
	// Verify nonce
	if ( ! check_ajax_referer( 'wz_review_nonce', 'nonce', false ) ) {
		wp_send_json_error( ['message' => 'Phiên làm việc đã hết hạn'] );
	}

	// Validate rating
	$rating = absint( $_POST['rating'] ?? 0 );
	if ( $rating < 1 || $rating > 5 ) {
		wp_send_json_error( ['message' => 'Đánh giá không hợp lệ'] );
	}

	// Validate content
	$content = sanitize_textarea_field( $_POST['content'] ?? '' );
	if ( strlen( $content ) < 50 ) {
		wp_send_json_error( ['message' => 'Nội dung phải có ít nhất 50 ký tự'] );
	}

	$product_id = absint( $_POST['product_id'] ?? 0 );
	$title = sanitize_text_field( $_POST['title'] ?? '' );

	// Get user info
	if ( is_user_logged_in() ) {
		$user = wp_get_current_user();
		$name = $user->display_name;
		$email = $user->user_email;
	} else {
		$name = sanitize_text_field( $_POST['name'] ?? '' );
		$email = sanitize_email( $_POST['email'] ?? '' );

		if ( empty( $name ) || empty( $email ) ) {
			wp_send_json_error( ['message' => 'Vui lòng điền đầy đủ thông tin'] );
		}
	}

	// Insert comment
	$comment_data = [
		'comment_post_ID' => $product_id,
		'comment_author' => $name,
		'comment_author_email' => $email,
		'comment_content' => $content,
		'comment_type' => 'review',
		'comment_approved' => 0, // Pending approval
	];

	$comment_id = wp_insert_comment( $comment_data );

	if ( $comment_id ) {
		// Save rating
		add_comment_meta( $comment_id, 'rating', $rating );
		add_comment_meta( $comment_id, 'title', $title );

		// Handle image uploads
		if ( ! empty( $_FILES['images'] ) ) {
			// Handle file uploads (simplified)
			// In production, use proper file validation
		}

		wp_send_json_success( ['message' => 'Cảm ơn bạn đã đánh giá! Đánh giá sẽ được hiển thị sau khi được duyệt.'] );
	} else {
		wp_send_json_error( ['message' => 'Không thể lưu đánh giá'] );
	}
}
```
