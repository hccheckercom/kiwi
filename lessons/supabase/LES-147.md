## Bad
```ts
export async function GET(req: Request) {
  const data = await fetchSensitiveData()
  return Response.json({ data }, { headers: corsHeaders() })
  // corsHeaders() without req → defaults to 'https://wezone.vn' only
  // But doesn't check actual request Origin → may block legitimate demo.wezone.vn requests
}
```

## Good
```ts
export async function GET(req: Request) {
  const data = await fetchSensitiveData()
  return Response.json({ data }, { headers: corsHeaders(req) })
  // corsHeaders(req) checks Origin header against allowlist:
  // - https://wezone.vn
  // - https://demo.wezone.vn
  // - https://admin-ec.wezone.vn
  // - localhost (dev)
  // Returns matching origin or omits header (browser blocks)
}
```

## OPTIONS handler
```ts
export function OPTIONS(req: Request) {
  return handleOptions(req) // Always pass req for proper Origin checking
}
```

## Why
Platform API is called from multiple domains (wezone.vn, demo.wezone.vn, localhost). corsHeaders(req) validates the Origin header against an allowlist and returns the matching origin. Without req, it falls back to a single hardcoded origin — breaking cross-origin requests from legitimate domains.
