## Bad (Next.js 15 pattern — breaks in 16)
```ts
export async function GET(req: Request, context: { params: { storeId: string } }) {
  const storeId = context.params.storeId // TypeError in Next.js 16!
}
```

## Good (Next.js 16 pattern)
```ts
export async function GET(
  req: NextRequest,
  context: { params: Promise<{ storeId: string }> }
) {
  const { storeId } = await context.params // Must await!
}
```

## Also in withPlatformKey middleware
```ts
export function withPlatformKey(handler: RouteHandler) {
  return async (req: Request, context?: { params?: Record<string, string> | Promise<Record<string, string>> }) => {
    const rawParams = context?.params
    // Handle both sync (legacy) and async (Next.js 16) params
    const resolvedParams = rawParams
      ? (rawParams instanceof Promise ? await rawParams : rawParams)
      : undefined
    return handler(req, store, resolvedParams)
  }
}
```

## Why
Next.js 16 changed route handler params from sync object to Promise. Without await, you get `[object Promise]` as the param value or TypeError. The withPlatformKey middleware handles both patterns for backward compatibility. All new routes must use `Promise<{ paramName: string }>` type.
