## Bad
```ts
const store = new Map<string, { count: number; windowStart: number }>()

export function checkRateLimit(key: string, limit: number): boolean {
  const entry = store.get(key)
  // Map grows forever — every unique IP adds an entry that never gets cleaned
  if (!entry) { store.set(key, { count: 1, windowStart: Date.now() }); return true }
  if (entry.count >= limit) return false
  entry.count++
  return true
}
```

## Good (pattern from lib/rate-limit.ts)
```ts
const MAX_STORE_SIZE = 10_000
const store = new Map<string, RateLimitEntry>()

// Cleanup interval — globalThis to survive HMR in dev
const _global = globalThis as typeof globalThis & { __rl_interval?: ReturnType<typeof setInterval> }
if (_global.__rl_interval) clearInterval(_global.__rl_interval)
_global.__rl_interval = setInterval(() => {
  const now = Date.now()
  for (const [key, entry] of store) {
    if (now - entry.windowStart > entry.windowMs) store.delete(key)
  }
}, 5 * 60_000)

// Unref so interval doesn't prevent process exit
if (typeof _global.__rl_interval === 'object' && 'unref' in _global.__rl_interval) {
  (_global.__rl_interval as { unref: () => void }).unref()
}

export function checkRateLimit(key: string, limit: number, windowMs = 60_000): boolean {
  const now = Date.now()
  const entry = store.get(key)
  if (!entry || now - entry.windowStart > windowMs) {
    // Evict oldest if at capacity
    if (!entry && store.size >= MAX_STORE_SIZE) {
      const oldestKey = store.keys().next().value
      if (oldestKey !== undefined) store.delete(oldestKey)
    }
    store.set(key, { count: 1, windowStart: now, windowMs })
    return true
  }
  if (entry.count >= limit) return false
  entry.count++
  return true
}
```

## Key patterns
- `MAX_STORE_SIZE` cap prevents unbounded growth
- `setInterval` cleanup removes expired entries every 5 minutes
- `globalThis` storage survives HMR re-evaluation in dev
- `.unref()` prevents interval from keeping Node.js process alive
- Single-instance only — multi-instance needs Redis

## Why
Real pattern from lib/rate-limit.ts. In-process rate limiters use a Map that grows with every unique IP. Without cleanup interval and size cap, a DDoS with rotating IPs fills server memory. The globalThis pattern ensures the cleanup interval isn't duplicated on HMR in development.
