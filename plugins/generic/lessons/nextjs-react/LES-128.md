## Bad
```ts
export async function GET(req: Request) {
  try {
    await requireAdmin(req)
    const { data } = await supabase.from('items').select('*')
    return Response.json({ data })
  } catch (err) {
    return Response.json({ error: 'Internal error' }, { status: 500 })
  }
}
// requireAdmin() throws redirect() on auth fail — catch swallows it → 500 instead of redirect
```

## Good
```ts
export async function GET(req: Request) {
  try {
    await requireAdmin(req)
    const { data } = await supabase.from('items').select('*')
    return Response.json({ data })
  } catch (err) {
    if (err instanceof Error && err.message === 'NEXT_REDIRECT') throw err
    return Response.json({ error: 'Internal error' }, { status: 500 })
  }
}
```

## Best (isRedirectError helper)
```ts
import { isRedirectError } from 'next/dist/client/components/redirect-error'

catch (err) {
  if (isRedirectError(err)) throw err
  return Response.json({ error: 'Internal error' }, { status: 500 })
}
```

## Why
Real bug: commit e567a6c. Next.js redirect() and notFound() work by throwing special errors. If your catch block catches everything, these navigation errors become 500 responses. Always re-throw redirect/notFound errors.
