## Bad
```ts
// Config loaded from DB or file — no validation
const config = await fetchIndustryConfig(slug) as IndustryConfig
// If config is malformed: runtime crash deep in component tree
return <ProductGrid config={config.productCard} /> // TypeError: Cannot read 'productCard' of undefined
```

## Good (pattern from lib/industry/validator.ts)
```ts
import { validateConfig } from '@/lib/industry/validator'

const raw = await fetchIndustryConfig(slug)
const result = validateConfig(raw)

if (!result.success) {
  console.error('[Industry] Config invalid:', result.errors)
  // Show fallback UI or redirect — don't crash
  return <ConfigErrorPage errors={result.errors} />
}

// result.data is typed IndustryConfig — safe to use
return <ProductGrid config={result.data.productCard} />
```

## Validator pattern (Zod safeParse — never throws)
```ts
export function validateConfig(raw: unknown): ValidateConfigResult {
  const result = IndustryConfigSchema.safeParse(raw)
  if (result.success) return { success: true, data: result.data as IndustryConfig }

  const errors = result.error.issues.map(e => `[${e.path.join('.')}] ${e.message}`)
  return { success: false, errors }
}
```

## Why
Industry configs come from DB (admin-editable) or JSON files. Without runtime validation, a missing field or wrong type crashes the entire page with an unhelpful TypeError. Zod safeParse catches all issues upfront with clear error messages — never throws, always returns Result type.
