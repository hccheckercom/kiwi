## Bad (Next.js 15 API — breaks in 16)
```ts
import { revalidateTag, revalidatePath } from 'next/cache'

export async function POST(req: Request) {
  await updateData()
  revalidateTag('homepage')
  revalidatePath('/admin/homepage')
}
```

## Good (Next.js 16 API)
```ts
import { revalidateTag, revalidatePath } from 'next/cache'

export async function POST(req: Request) {
  await updateData()
  await revalidateTag('homepage')    // Now async in Next.js 16
  await revalidatePath('/admin/homepage')
}
```

## Why
Real bug: commit 26fed29. Next.js 16 changed revalidateTag/revalidatePath to async functions. Without await, revalidation doesn't complete before response is sent — stale data persists. Build doesn't warn, TypeScript doesn't catch it (returns Promise which is truthy).
