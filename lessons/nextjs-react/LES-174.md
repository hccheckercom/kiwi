## Bad
```ts
export async function fetchPage(url: string) {
  const response = await fetch(url)
  const html = await response.text() // 500MB page → OOM crash
  return parseHtml(html)
}
```

## Good (bounded fetch pattern)
```ts
export async function fetchPage(url: string) {
  const controller = new AbortController()
  const timeoutId = setTimeout(() => controller.abort(), 10_000)

  try {
    const response = await fetch(url, { signal: controller.signal })

    // Check Content-Length header first (fast reject)
    const contentLength = parseInt(response.headers.get('content-length') ?? '0')
    if (contentLength > 5_000_000) {
      return makeErrorResult(url, 'Page too large (>5MB)')
    }

    const html = await response.text()

    // Double-check actual size (Content-Length can lie)
    if (html.length > 5_000_000) {
      return makeErrorResult(url, 'Page too large (>5MB)')
    }

    return { html, httpStatus: response.status, ... }
  } finally {
    clearTimeout(timeoutId)
  }
}
```

## Why
Real fix from GMC audit (commit c71de28 "decompression bomb"). When fetching user-supplied URLs, a malicious site can return gigabytes of data (decompression bomb via gzip). Without size limits, Node.js runs out of memory and crashes the entire server. Always bound response size before calling .text().
