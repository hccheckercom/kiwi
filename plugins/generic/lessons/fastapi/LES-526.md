# Bad Code

```python
@app.get("/users/{user_id}")
async def get_user(user_id: int, db: Session = Depends(get_db)):
    user = db.query(User).filter(User.id == user_id).first()
    if not user:
        return {"error": "User not found"}  # BAD: 200 status with error
    return user

@app.delete("/users/{user_id}")
async def delete_user(user_id: int, db: Session = Depends(get_db)):
    user = db.query(User).filter(User.id == user_id).first()
    if not user:
        return None  # BAD: 200 status with null body
    db.delete(user)
    return {"message": "Deleted"}
```

# Good Code

```python
from fastapi import HTTPException, status

@app.get("/users/{user_id}")
async def get_user(user_id: int, db: Session = Depends(get_db)):
    user = db.query(User).filter(User.id == user_id).first()
    if not user:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="User not found"
        )
    return user

@app.delete("/users/{user_id}")
async def delete_user(user_id: int, db: Session = Depends(get_db)):
    user = db.query(User).filter(User.id == user_id).first()
    if not user:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="User not found"
        )
    db.delete(user)
    return {"message": "Deleted"}

# Custom error responses
@app.post("/users")
async def create_user(user: UserCreate, db: Session = Depends(get_db)):
    existing = db.query(User).filter(User.email == user.email).first()
    if existing:
        raise HTTPException(
            status_code=status.HTTP_409_CONFLICT,
            detail="Email already registered",
            headers={"X-Error-Code": "EMAIL_EXISTS"}
        )
    db_user = User(**user.dict())
    db.add(db_user)
    return db_user
```

# Why

**Returning error dicts is wrong:**
- Client gets 200 OK with error message
- Breaks HTTP semantics
- Clients can't distinguish success/failure
- Monitoring tools miss errors
- Retry logic doesn't trigger

**Correct HTTP status codes:**
```python
# 400 Bad Request - Invalid input
raise HTTPException(400, "Invalid email format")

# 401 Unauthorized - Not authenticated
raise HTTPException(401, "Authentication required")

# 403 Forbidden - Authenticated but not allowed
raise HTTPException(403, "Admin access required")

# 404 Not Found - Resource doesn't exist
raise HTTPException(404, "User not found")

# 409 Conflict - Resource already exists
raise HTTPException(409, "Email already registered")

# 422 Unprocessable Entity - Validation failed (Pydantic auto)
# FastAPI returns this automatically for validation errors

# 500 Internal Server Error - Unexpected error
# FastAPI returns this automatically for unhandled exceptions
```

**Error response format:**
```json
{
  "detail": "User not found"
}

// Or structured
{
  "detail": {
    "code": "USER_NOT_FOUND",
    "message": "User with ID 123 not found",
    "user_id": 123
  }
}
```

**Global exception handlers:**
```python
@app.exception_handler(ValueError)
async def value_error_handler(request, exc):
    return JSONResponse(
        status_code=400,
        content={"detail": str(exc)}
    )
```
