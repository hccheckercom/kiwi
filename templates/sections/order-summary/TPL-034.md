<?php
declare(strict_types=1);

if ( ! defined( 'ABSPATH' ) ) { exit; }

$phone = function_exists( 'wz_config' ) ? wz_config( 'phone', '' ) : '';
?>

<div class="lg:sticky lg:top-[120px] space-y-4">
    <div class="bg-[var(--wz-surface-pure)] rounded-[var(--wz-radius-card)] shadow-[0_4px_24px_rgba(27,58,122,0.08)] p-5">
        <h2 class="text-base font-bold text-[var(--wz-primary)] mb-4">Đơn hàng của bạn</h2>

        <!-- Mini product list -->
        <div class="space-y-2 mb-4">
            <?php $shown = 0; foreach ( $cart_items as $item ) :
                if ( $shown >= 3 ) { break; }
                $pid     = (int) $item['product_id'];
                $qty     = (int) ( $item['qty'] ?? $item['quantity'] ?? 1 );
                $product = function_exists( 'wz_get_product' ) ? wz_get_product( $pid ) : null;
                if ( ! $product ) { continue; }
                $images = function_exists( 'wz_get_product_images' ) ? wz_get_product_images( $pid ) : [];
                $thumb  = ! empty( $images ) ? $images[0]['thumbnail'] : '';
                $shown++;
            ?>
                <div class="flex items-center gap-2">
                    <div class="w-10 h-10 rounded overflow-hidden bg-[var(--wz-surface-cool)] shrink-0">
                        <?php if ( $thumb ) : ?>
                            <img src="<?php echo esc_url( $thumb ); ?>" alt="<?php echo esc_attr( $product['name'] ); ?>" class="w-full h-full object-cover" loading="lazy">
                        <?php endif; ?>
                    </div>
                    <div class="flex-1 min-w-0">
                        <p class="text-xs text-[var(--wz-on-surface)] truncate"><?php echo esc_html( $product['name'] ); ?></p>
                    </div>
                    <span class="text-xs text-[var(--wz-outline)] shrink-0">x<?php echo $qty; ?></span>
                </div>
            <?php endforeach; ?>
            <?php if ( $cart_count > 3 ) : ?>
                <p class="text-xs text-[var(--wz-outline)]">+<?php echo esc_html( (string) ( $cart_count - 3 ) ); ?> sản phẩm khác</p>
            <?php endif; ?>
        </div>

        <!-- Totals -->
        <div class="space-y-2 text-sm border-t border-[var(--wz-outline)]/10 pt-4">
            <div class="flex justify-between">
                <span class="text-[var(--wz-outline)]">Tạm tính</span>
                <span class="font-semibold"><?php echo esc_html( wz_format_price( $subtotal ) ); ?></span>
            </div>
            <div class="flex justify-between">
                <span class="text-[var(--wz-outline)]">Phí vận chuyển</span>
                <span class="font-semibold <?php echo $shipping_free ? 'text-[var(--wz-accent)]' : ''; ?>">
                    <?php echo $shipping_free ? 'Miễn phí' : esc_html( wz_format_price( $shipping_cost ) ); ?>
                </span>
            </div>
            <div class="flex justify-between border-t border-[var(--wz-outline)]/10 pt-2">
                <span class="font-bold text-[var(--wz-on-surface)]">Tổng cộng</span>
                <span class="text-xl font-bold text-[var(--wz-sale-price)]"><?php echo esc_html( wz_format_price( $total ) ); ?></span>
            </div>
        </div>

        <!-- Place order button -->
        <button type="submit" id="wz-place-order"
                class="mt-5 w-full flex items-center justify-center gap-2 py-3 bg-[var(--wz-primary)] text-white rounded-[var(--wz-radius)] font-semibold text-sm hover:brightness-110 transition">
            <span class="material-symbols-outlined text-[20px]" aria-hidden="true">check_circle</span>
            Đặt hàng
        </button>

        <!-- Legal -->
        <p class="text-[10px] text-[var(--wz-outline)] text-center mt-3 leading-relaxed">
            Bằng việc đặt hàng, bạn đồng ý với
            <a href="<?php echo esc_url( home_url( '/dieu-khoan-su-dung/' ) ); ?>" class="underline hover:text-[var(--wz-primary)]">Điều khoản dịch vụ</a> và
            <a href="<?php echo esc_url( home_url( '/chinh-sach-bao-mat/' ) ); ?>" class="underline hover:text-[var(--wz-primary)]">Chính sách bảo mật</a>
        </p>
    </div>

    <!-- Support -->
    <?php if ( $phone ) : ?>
    <div class="bg-[var(--wz-surface-cool)] rounded-[var(--wz-radius-card)] p-4 text-center">
        <p class="text-xs text-[var(--wz-outline)]">Cần hỗ trợ? Gọi ngay</p>
        <a href="tel:<?php echo esc_attr( preg_replace( '/[^0-9]/', '', $phone ) ); ?>" class="text-sm font-bold text-[var(--wz-primary)]"><?php echo esc_html( $phone ); ?></a>
    </div>
    <?php endif; ?>
</div>

<?php
declare(strict_types=1);

if ( ! defined( 'ABSPATH' ) ) { exit; }

$show_count = 3;
$hidden_count = max( 0, $cart_count - $show_count );
?>

<div class="bg-[var(--wz-surface-pure)] rounded-[var(--wz-radius-card)] shadow-[0_4px_24px_rgba(27,58,122,0.08)] p-5 md:p-6">
    <h2 class="text-base font-bold text-[var(--wz-primary)] mb-4 flex items-center gap-2">
        <span class="material-symbols-outlined text-[20px]" aria-hidden="true">inventory_2</span>
        Sản phẩm trong đơn (<?php echo esc_html( (string) $cart_count ); ?>)
    </h2>

    <div class="space-y-3">
        <?php $i = 0; foreach ( $cart_items as $item ) :
            $pid     = (int) $item['product_id'];
            $qty     = (int) ( $item['qty'] ?? 1 );
            $product = function_exists( 'wz_get_product' ) ? wz_get_product( $pid ) : null;
            if ( ! $product ) { continue; }
            $images = function_exists( 'wz_get_product_images' ) ? wz_get_product_images( $pid ) : [];
            $thumb  = ! empty( $images ) ? $images[0]['thumbnail'] : '';
            $i++;
        ?>
            <div class="flex items-center gap-3 <?php echo $i > $show_count ? 'wz-checkout-item-hidden hidden' : ''; ?>">
                <div class="w-12 h-12 rounded-[var(--wz-radius)] overflow-hidden bg-[var(--wz-surface-cool)] shrink-0">
                    <?php if ( $thumb ) : ?>
                        <img src="<?php echo esc_url( $thumb ); ?>" alt="<?php echo esc_attr( $product['name'] ); ?>" class="w-full h-full object-cover" loading="lazy">
                    <?php else : ?>
                        <span class="material-symbols-outlined text-[20px] text-[var(--wz-outline)]/30 flex items-center justify-center w-full h-full" aria-hidden="true">image</span>
                    <?php endif; ?>
                </div>
                <div class="flex-1 min-w-0">
                    <p class="text-sm font-medium text-[var(--wz-on-surface)] line-clamp-2"><?php echo esc_html( $product['name'] ); ?></p>
                    <?php if ( ! empty( $item['variation'] ) ) : ?>
                        <p class="text-xs text-[var(--wz-outline)]"><?php echo esc_html( $item['variation'] ); ?></p>
                    <?php endif; ?>
                </div>
                <div class="text-right shrink-0">
                    <p class="text-xs text-[var(--wz-outline)]">x<?php echo esc_html( (string) $qty ); ?></p>
                    <p class="text-sm font-bold text-[var(--wz-primary)]"><?php echo esc_html( wz_format_price( (float) $product['price'] * $qty ) ); ?></p>
                </div>
            </div>
        <?php endforeach; ?>
    </div>

    <?php if ( $hidden_count > 0 ) : ?>
        <button type="button" id="wz-checkout-expand-items" class="mt-3 text-xs font-semibold text-[var(--wz-primary)] hover:underline">
            Xem thêm <?php echo esc_html( (string) $hidden_count ); ?> sản phẩm
        </button>
    <?php endif; ?>
</div>
