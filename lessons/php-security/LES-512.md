## Bad
```php
public function check_redirect(): void {
    if ( ! is_404() ) {
        return;
    }
    
    global $wpdb;
    $table   = $wpdb->prefix . 'wezone_migrator_redirects';
    $request = sanitize_text_field( wp_unslash( $_SERVER['REQUEST_URI'] ?? '' ) );
    $path    = wp_parse_url( $request, PHP_URL_PATH );
    
    if ( ! $path ) {
        return;
    }
    
    // Query DB on every 404 â€” no caching
    $redirect = $wpdb->get_row(
        $wpdb->prepare( "SELECT * FROM {$table} WHERE old_url = %s", $path )
    );
    
    if ( $redirect ) {
        wp_safe_redirect( $redirect->new_url, 301 );
        exit;
    }
}
```

## Good
```php
public function check_redirect(): void {
    if ( ! is_404() ) {
        return;
    }
    
    global $wpdb;
    $table   = $wpdb->prefix . 'wezone_migrator_redirects';
    $request = sanitize_text_field( wp_unslash( $_SERVER['REQUEST_URI'] ?? '' ) );
    $path    = wp_parse_url( $request, PHP_URL_PATH );
    
    if ( ! $path ) {
        return;
    }
    
    // Cache redirect lookups for 1 hour
    $cache_key = 'wz_redirect_' . md5( $path );
    $redirect  = wp_cache_get( $cache_key, 'wezone_migrator' );
    
    if ( false === $redirect ) {
        $redirect = $wpdb->get_row(
            $wpdb->prepare( "SELECT * FROM {$table} WHERE old_url = %s", $path )
        );
        wp_cache_set( $cache_key, $redirect ?: 'not_found', 'wezone_migrator', HOUR_IN_SECONDS );
    }
    
    if ( $redirect && 'not_found' !== $redirect ) {
        $wpdb->query(
            $wpdb->prepare( "UPDATE {$table} SET hits = hits + 1 WHERE id = %d", $redirect->id )
        );
        wp_safe_redirect( $redirect->new_url, 301 );
        exit;
    }
}
```

## Why
Attacker floods 404 URLs â†’ DB query on every request â†’ table scan â†’ CPU spike â†’ site down. Must cache redirect lookups.
