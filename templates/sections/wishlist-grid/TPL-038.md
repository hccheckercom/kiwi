## Code — PHP

```php
<!-- wishlist.php -->
<?php
declare(strict_types=1);

if ( ! defined( 'ABSPATH' ) ) { exit; }

if ( ! is_user_logged_in() ) {
    wp_safe_redirect( home_url( '/my-account/login/' ) );
    exit;
}

get_header();

$user_id  = get_current_user_id();
$wishlist = get_user_meta( $user_id, 'wz_wishlist', true );
$wishlist = is_array( $wishlist ) ? $wishlist : [];

set_query_var( 'wz_tpl_account_page', 'wishlist' );
?>

<div class="bg-[var(--wz-surface)] min-h-screen">
<div class="max-w-[1280px] mx-auto px-5 md:px-[20px] py-4">

    <nav aria-label="Breadcrumb" class="pb-4">
        <ol class="flex items-center gap-1 text-xs md:text-sm text-[var(--wz-outline)]">
            <li><a href="<?php echo esc_url( home_url( '/' ) ); ?>" class="hover:text-[var(--wz-primary)] transition">Trang chủ</a></li>
            <li><span class="material-symbols-outlined text-[14px]" aria-hidden="true">chevron_right</span></li>
            <li><a href="<?php echo esc_url( home_url( '/my-account/' ) ); ?>" class="hover:text-[var(--wz-primary)] transition">Tài khoản</a></li>
            <li><span class="material-symbols-outlined text-[14px]" aria-hidden="true">chevron_right</span></li>
            <li class="font-bold text-[var(--wz-primary)]" aria-current="page">Yêu thích</li>
        </ol>
    </nav>

    <div class="flex flex-col lg:flex-row lg:gap-6">
        <?php include get_stylesheet_directory() . '/template-parts/account/sidebar.php'; ?>

        <div class="flex-1 min-w-0 mt-4 lg:mt-0 space-y-4">
            <h1 class="text-xl font-bold text-[var(--wz-primary)]">Sản phẩm yêu thích (<?php echo count( $wishlist ); ?>)</h1>

            <?php if ( empty( $wishlist ) ) : ?>
                <div class="bg-[var(--wz-surface-pure)] rounded-[var(--wz-radius-card)] p-8 shadow-[0_4px_24px_rgba(27,58,122,0.08)] text-center">
                    <span class="material-symbols-outlined text-[48px] text-[var(--wz-outline)]/30" aria-hidden="true">favorite_border</span>
                    <p class="text-base font-bold text-[var(--wz-primary)] mt-3">Chưa có sản phẩm yêu thích</p>
                    <p class="text-sm text-[var(--wz-outline)] mt-1">Thêm sản phẩm vào danh sách yêu thích để mua sau</p>
                    <a href="<?php echo esc_url( function_exists( 'wz_get_shop_url' ) ? wz_get_shop_url() : home_url( '/shop/' ) ); ?>" class="inline-flex items-center gap-2 mt-4 px-5 py-2 bg-[var(--wz-primary)] text-white rounded-[var(--wz-radius)] text-sm font-semibold hover:brightness-110 transition">Khám phá sản phẩm</a>
                </div>
            <?php else : ?>
                <div class="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">
                    <?php foreach ( $wishlist as $product_id ) :
                        get_template_part( 'template-parts/product/card', null, [
                            'product_id' => (int) $product_id,
                        ] );
                    endforeach; ?>
                </div>
            <?php endif; ?>
        </div>
    </div>

</div>
</div>

<?php get_footer(); ?>
```
