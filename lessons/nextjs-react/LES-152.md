## Bad
```ts
export async function getAllProviders(): Promise<AIProviderConfig[]> {
  try {
    const { data, error } = await supabase.from('ai_providers').select('*')
    if (error) return [] // Empty array — all AI features silently break
  } catch {
    return [] // Same problem
  }
}
```

## Good (pattern from lib/ai/provider-manager.ts)
```ts
let cachedProviders: AIProviderConfig[] | null = null
let cacheTimestamp = 0
const CACHE_TTL = 5 * 60 * 1000

export async function getAllProviders(): Promise<AIProviderConfig[]> {
  const now = Date.now()
  if (cachedProviders && (now - cacheTimestamp) < CACHE_TTL) {
    return cachedProviders // Serve from cache — fast path
  }

  try {
    const { data, error } = await supabase.from('ai_providers').select('*')
    if (error) {
      console.error('[AI Provider Manager] DB error:', error.message)
      return cachedProviders ?? [] // Fallback to stale cache, not empty
    }
    cachedProviders = data.map(mapRowToConfig)
    cacheTimestamp = now
    return cachedProviders
  } catch (err) {
    console.error('[AI Provider Manager] Failed to load:', err)
    return cachedProviders ?? [] // Graceful degradation
  }
}
```

## Key pattern: invalidateProviderCache()
```ts
// After admin updates provider config, invalidate cache
export function invalidateProviderCache() {
  cachedProviders = null
  cacheTimestamp = 0
}
```

## Why
Real pattern from provider-manager.ts. AI providers are fetched from DB but cached in-memory (5min TTL). On DB error, returning empty array breaks all AI features. Returning stale cache keeps features working with last-known-good config. Always provide invalidation function for admin updates.
