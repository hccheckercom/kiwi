## Bad
```ts
'use server'
export async function updateProduct(id: string, data: ProductData) {
  await supabase.from('products').update(data).eq('id', id)
  return { success: true }
}
// Page still shows old data until manual refresh
```

## Good
```ts
'use server'
import { revalidatePath } from 'next/cache'

export async function updateProduct(id: string, data: ProductData) {
  await supabase.from('products').update(data).eq('id', id)
  revalidatePath('/admin/products')
  return { success: true }
}
```

## Why
Next.js caches page data aggressively. After a server action mutates data, the cached page still shows stale content. Always call `revalidatePath` or `revalidateTag` after mutations to bust the cache and show fresh data.
