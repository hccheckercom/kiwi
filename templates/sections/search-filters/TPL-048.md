<?php
declare(strict_types=1);

if ( ! defined( 'ABSPATH' ) ) { exit; }

add_action( 'wp_ajax_wz_live_search', 'wezone_funilux_live_search' );
add_action( 'wp_ajax_nopriv_wz_live_search', 'wezone_funilux_live_search' );

function wezone_funilux_live_search(): void {
    check_ajax_referer( 'wz-theme-nonce', 'nonce' );

    $keyword = sanitize_text_field( wp_unslash( $_GET['q'] ?? '' ) );

    if ( mb_strlen( $keyword ) < 2 ) {
        wp_send_json_success( [ 'products' => [], 'total' => 0, 'keyword' => $keyword ] );
    }

    $results = wz_search_products_db( $keyword, 6 );

    wp_send_json_success( [
        'products' => $results['items'],
        'total'    => $results['total'],
        'keyword'  => $keyword,
    ] );
}

if ( ! function_exists( 'wz_search_products_db' ) ) {
function wz_search_products_db( string $keyword, int $limit = 12, int $offset = 0, string $orderby = '' ): array {
    global $wpdb;
    $table = $wpdb->prefix . 'wz_products';

    $table_exists = $wpdb->get_var( "SHOW TABLES LIKE '{$table}'" );
    if ( ! $table_exists ) {
        return [ 'items' => [], 'total' => 0 ];
    }

    $like = '%' . $wpdb->esc_like( $keyword ) . '%';

    $count = (int) $wpdb->get_var( $wpdb->prepare(
        "SELECT COUNT(*) FROM {$table} WHERE status = 'publish' AND (name LIKE %s OR sku LIKE %s)",
        $like, $like
    ) );

    $order_clause = 'ORDER BY id DESC';
    switch ( $orderby ) {
        case 'price_asc':
            $order_clause = 'ORDER BY COALESCE(NULLIF(sale_price, 0), regular_price) ASC';
            break;
        case 'price_desc':
            $order_clause = 'ORDER BY COALESCE(NULLIF(sale_price, 0), regular_price) DESC';
            break;
        case 'date':
            $order_clause = 'ORDER BY created_at DESC';
            break;
        case 'name':
            $order_clause = 'ORDER BY name ASC';
            break;
    }

    $rows = $wpdb->get_results( $wpdb->prepare(
        "SELECT * FROM {$table} WHERE status = 'publish' AND (name LIKE %s OR sku LIKE %s) {$order_clause} LIMIT %d OFFSET %d",
        $like, $like, $limit, $offset
    ), ARRAY_A );

    $items = [];
    foreach ( $rows as $row ) {
        $regular = (float) ( $row['regular_price'] ?? 0 );
        $sale    = (float) ( $row['sale_price'] ?? 0 );
        $price   = $sale > 0 ? $sale : $regular;
        $slug    = $row['slug'] ?? '';
        $img_id  = (int) ( $row['image_id'] ?? 0 );
        $image   = $img_id ? wp_get_attachment_image_url( $img_id, 'thumbnail' ) : '';

        $price_html = '';
        if ( function_exists( 'wz_format_price' ) && $price > 0 ) {
            $price_html = wz_format_price( $price );
        } elseif ( $price > 0 ) {
            $price_html = number_format( $price, 0, ',', '.' ) . '₫';
        }

        $items[] = [
            'id'    => (int) $row['id'],
            'name'  => $row['name'] ?? '',
            'url'   => $slug ? home_url( '/product/' . $slug . '/' ) : '#',
            'image' => $image ?: '',
            'price' => $price_html,
        ];
    }

    return [ 'items' => $items, 'total' => $count ];
}
}
